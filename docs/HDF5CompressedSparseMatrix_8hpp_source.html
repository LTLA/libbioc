70 auto check_size = [&](const H5::DataSet& handle, const std::string& name) -> size_t {
71 auto type = handle.getTypeClass();
72 if (type != H5T_INTEGER && type != H5T_FLOAT) {
73 throw std::runtime_error(std::string("expected numeric values in the '") + name + "' dataset");
74 }
75 auto space = handle.getSpace();
76
77 int ndim = space.getSimpleExtentNdims();
78 if (ndim != 1) {
79 throw std::runtime_error(std::string("'") + name + "' should be a one-dimensional array");
80 }
81
82 hsize_t dims_out[1];
83 space.getSimpleExtentDims(dims_out, NULL);
84 return dims_out[0];
85 };
86
87 auto dhandle = file_handle.openDataSet(data_name);
88 const size_t nonzeros = check_size(dhandle, "vals");
89
90 auto ihandle = file_handle.openDataSet(index_name);
91 if (check_size(ihandle, "idx") != nonzeros) {
92 throw std::runtime_error("number of non-zero elements is not consistent between 'data' and 'idx'");
93 }
94
95 auto phandle = file_handle.openDataSet(ptr);
96 const size_t ptr_size = check_size(phandle, "ptr");
97 if (ptr_size != pointers.size()) {
98 throw std::runtime_error("'ptr' is not of the appropriate length");
99 }
100
101 // Checking the contents of the index pointers.
102 phandle.read(pointers.data(), H5::PredType::NATIVE_HSIZE);
103 if (pointers[0] != 0) {
104 throw std::runtime_error("first index pointer should be zero");
105 }
106 if (pointers.back() != nonzeros) {
107 throw std::runtime_error("last index pointer should be equal to the number of non-zero elements");
108 }
109
110 size_t maxgap = 0;
111 for (size_t i = 1; i < pointers.size(); ++i) {
112 if (pointers[i] < pointers[i-1]) {
113 throw std::runtime_error("index pointers should be sorted");
114 }
115 size_t delta = pointers[i] - pointers[i - 1];
116 if (delta > maxgap) {
117 maxgap = delta;
118 }
119 }
120
121 // Setting up the cache parameters.
122 cache_id.resize(pointers.size() - 1);
123 if (cache_id.size()) {
124 cache_limits.emplace_back(0, pointers[1]);
125 }
126
127 size_t effective_cache_limit = cache_limit / (dhandle.getDataType().getSize() + ihandle.getDataType().getSize());
128
129 size_t counter = 0, start = 0;
130 for (size_t i = 1; i < cache_id.size(); ++i) {
131 size_t end = pointers[i + 1];
132 if (end - start <= effective_cache_limit) {
133 cache_limits.back().second = end;
134 } else {
135 ++counter;
136 start = pointers[i];
137 cache_limits.emplace_back(start, end);
138 }
139 cache_id[i] = counter;
140 }
141
142 return;
143 }

205 template<typename IndexIt, typename DataIt, typename Thing>
206 size_t copy_primary_to_buffer(IndexIt istart, DataIt dstart, size_t len, size_t first, size_t last, T* dbuffer, Thing thing) const {
207 size_t request_start = (first > *istart ? std::lower_bound(istart, istart + len, first) - istart : 0);
208 size_t request_end = (last < *(istart + len - 1) ? std::lower_bound(istart + request_start, istart + len, last) - istart : len);
209
210 if constexpr(std::is_same<typename std::remove_reference<Thing>::type, IDX*>::value) {
211 std::copy(istart + request_start, istart + request_end, thing);
212 std::copy(dstart + request_start, dstart + request_end, dbuffer);
213 } else {
214 std::fill(dbuffer, dbuffer + (last - first), 0);
215 for (size_t i = request_start; i < request_end; ++i) {
216 dbuffer[*(istart + i) - first] = *(dstart + i);
217 }
218 }
219
220 return request_end - request_start; // number of non-zeros.
221 }

224 size_t extract_primary(size_t i, T* dbuffer, Thing thing, size_t first, size_t last, HDF5SparseWorkspace& work) const {
225 if (cache_id[i] != work.current_cache_id) {
226 // Pulling out the entire chunk containing 'i'.
227 work.current_cache_id = cache_id[i];
228 const auto& limits = cache_limits[work.current_cache_id];
229 hsize_t offset = limits.first;
230 hsize_t count = limits.second - limits.first;
231
232 work.dataspace.selectHyperslab(H5S_SELECT_SET, &count, &offset);
233 work.memspace.setExtentSimple(1, &count);
234 work.memspace.selectAll();
235
236 work.index_cache.resize(count);
237 work.index.read(work.index_cache.data(), HDF5::define_mem_type<IDX>(), work.memspace, work.dataspace);
238
239 work.data_cache.resize(count);
240 work.data.read(work.data_cache.data(), HDF5::define_mem_type<T>(), work.memspace, work.dataspace);
241 }
242
243 size_t len = pointers[i + 1] - pointers[i];
244 if (len == 0) {
245 return 0;
246 }
247
248 const auto& limits = cache_limits[work.current_cache_id];
249 size_t offset = pointers[i] - limits.first;
250 auto istart = work.index_cache.begin() + offset;
251 auto dstart = work.data_cache.begin() + offset;
252 return copy_primary_to_buffer(istart, dstart, len, first, last, dbuffer, thing);
253 }

294 template<typename Thing>
295 size_t extract_secondary(size_t i, T* dbuffer, Thing thing, size_t first, size_t last,
296 H5::DataSet& data, H5::DataSet& index,
297 H5::DataSpace& dataspace, H5::DataSpace& memspace,
298 std::vector<IDX>& index_cache) const
299 {
300 // Looping over all secondary dimensions.
301 size_t counter = 0;
302 for (size_t j = first; j < last; ++j) {
303 size_t left = pointers[j], right = pointers[j + 1];
304 index_cache.resize(right - left);
305
306 hsize_t offset = left;
307 hsize_t count = index_cache.size();
308 dataspace.selectHyperslab(H5S_SELECT_SET, &count, &offset);
309 memspace.setExtentSimple(1, &count);
310 memspace.selectAll();
311 index.read(index_cache.data(), HDF5::define_mem_type<IDX>(), memspace, dataspace);
312
313 auto it = std::lower_bound(index_cache.begin(), index_cache.end(), i);
314 if (it != index_cache.end() && *it == i) {
315 offset = left + (it - index_cache.begin());
316 count = 1;
317 dataspace.selectHyperslab(H5S_SELECT_SET, &count, &offset);
318 memspace.setExtentSimple(1, &count);
319 memspace.selectAll();
320
321 auto dest = dbuffer;
322 if constexpr(std::is_same<Thing, IDX*>::value) {
323 dest += counter;
324 } else {
325 dest += j - first;
326 }
327 data.read(dest, HDF5::define_mem_type<T>(), memspace, dataspace);
328
329 if constexpr(std::is_same<typename std::remove_reference<Thing>::type, IDX*>::value) {
330 thing[counter] = j;
331 ++counter;
332 }
333 }
334 }
335
336 return counter;
337 }
