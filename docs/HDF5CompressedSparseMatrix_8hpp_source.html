219 void populate_primary_cache(size_t i, PrimaryH5Core& core, bool needs_value) const {
220 if (primary_cache_id[i] == core.current_cache_id && core.init) {
221 return;
222 }
223
224 core.init = true;
225 core.current_cache_id = primary_cache_id[i];
226
227 // Pulling out the entire chunk of primary dimensions containing
228 // 'i'. We have to do this for all indices, regardless of the
229 // slicing/indexing.
230 const auto& limits = primary_cache_limits[core.current_cache_id];
231 hsize_t offset = limits.first;
232 hsize_t count = limits.second - limits.first;
233
234#ifndef TATAMI_HDF5_PARALLEL_LOCK
235 #pragma omp critical
236 {
237#else
238 TATAMI_HDF5_PARALLEL_LOCK([&]() -> void {
239#endif

241 core.dataspace.selectHyperslab(H5S_SELECT_SET, &count, &offset);
242 core.memspace.setExtentSimple(1, &count);
243 core.memspace.selectAll();
244
245 core.index_cache.resize(count);
246 core.index.read(core.index_cache.data(), HDF5::define_mem_type<Index_>(), core.memspace, core.dataspace);
247
248 if (needs_value) {
249 // In theory, we could avoid extracting data for the entire column when
250 // populating the cache for sliced or indexed queries. In practice,
251 // each chunk is often larger than the number of non-zeroes in a
252 // column, so we end up having to pull out the entire column anyway.
253 // Also, I want to get out of this critical region ASAP and do my
254 // indexing elsewhere.

256 core.data_cache.resize(count);

269 void extract_primary_raw(size_t i, Function_ fill, Index_ start, Index_ length, PrimaryH5Core& core, bool needs_value) const {
270 if (length == 0) {
271 return;
272 }
273
274 Index_ primary_length = pointers[i + 1] - pointers[i];
275 if (primary_length ==0) {
276 return;
277 }
278
279 populate_primary_cache(i, core, needs_value);
280
281 const auto& limits = primary_cache_limits[core.current_cache_id];
282 Index_ offset = pointers[i] - limits.first;
283 auto istart = core.index_cache.begin() + offset;
284
285 Index_ request_start = 0;
286 if (start > *istart) { // 'istart' guaranteed to be valid from length check above.
287 bool do_cache = !core.starts.empty();
288 if (do_cache && core.starts[i] != -1) {
289 request_start = core.starts[i];
290 } else {
291 request_start = std::lower_bound(istart, istart + primary_length, start) - istart;
292 if (do_cache) {
293 core.starts[i] = request_start;
294 }
295 }
296 }
297
298 istart += request_start;
299 Index_ end = start + length;
300 if (needs_value) {
301 auto dstart = core.data_cache.begin() + offset + request_start;
302 for (size_t i = request_start; i < primary_length && *istart < end; ++i, ++istart, ++dstart) {
303 fill(*istart, *dstart);
304 }
305 } else {
306 for (size_t i = request_start; i < primary_length && *istart < end; ++i, ++istart) {
307 fill(*istart, 0);
308 }
309 }
310
311 return;

360 void extract_primary_raw(size_t i, Function_ fill, Skip_ skip, const std::vector<Index_>& indices, PrimaryH5Core& core, bool needs_value) const {
361 if (indices.empty()) {
362 return;
363 }
364
365 Index_ primary_length = pointers[i + 1] - pointers[i];
366 if (primary_length ==0) {
367 return;
368 }
369
370 populate_primary_cache(i, core, needs_value);
371
372 const auto& limits = primary_cache_limits[core.current_cache_id];
373 Index_ offset = pointers[i] - limits.first;
374 auto istart = core.index_cache.begin() + offset;
375 auto iend = istart + primary_length;
376
377 Index_ quick_shift = 0;
378 if (indices[0] > *istart) { // Both are guaranteed to valid, from length check above.
379 bool do_cache = !core.starts.empty();
380 if (do_cache && core.starts[i] != -1) {
381 quick_shift = core.starts[i];
382 } else {
383 quick_shift = std::lower_bound(istart, istart + primary_length, indices[0]) - istart;
384 if (do_cache) {
385 core.starts[i] = quick_shift;
386 }
387 }
388 }
389
390 istart += quick_shift;
391 auto dstart = core.data_cache.begin();
392 if (needs_value) {
393 dstart += offset + quick_shift;
394 }
395
396 for (auto idx : indices) {
397 while (istart != iend && *istart < idx) {
398 ++istart;
399 ++dstart;
400 }
401 if (istart == iend) {
402 break;
403 }
404 if (*istart == idx) {
405 if (needs_value) {
406 fill(idx, *dstart);
407 } else {
408 fill(idx, 0);
409 }
410 } else {
411 skip();
412 }
413 }

469 // This could be improved by extracting multiple rows at any given call and
470 // caching them for subsequent requests. However, even then, we'd require
471 // multiple re-reads from file when we exceed the cache. So, any caching
472 // would be just turning an extremely bad access pattern into a very bad
473 // pattern, when users shouldn't even be calling this at all...
474
475 template<class Function_>
476 bool extract_secondary_raw(Index_ primary, Index_ secondary, Function_& fill, H5Core& core, bool needs_value) const {
477 hsize_t left = pointers[primary], right = pointers[primary + 1];
478 core.index_cache.resize(right - left);
479
480 // Serial locks should be applied by the callers.
481 hsize_t offset = left;
482 hsize_t count = core.index_cache.size();
483 core.dataspace.selectHyperslab(H5S_SELECT_SET, &count, &offset);
484 core.memspace.setExtentSimple(1, &count);
485 core.memspace.selectAll();
486 core.index.read(core.index_cache.data(), HDF5::define_mem_type<Index_>(), core.memspace, core.dataspace);
487
488 auto it = std::lower_bound(core.index_cache.begin(), core.index_cache.end(), secondary);
489 if (it != core.index_cache.end() && *it == secondary) {
490 if (needs_value) {
491 offset = left + (it - core.index_cache.begin());
492 count = 1;
493 core.dataspace.selectHyperslab(H5S_SELECT_SET, &count, &offset);
494 core.memspace.setExtentSimple(1, &count);
495 core.memspace.selectAll();
496
497 Value_ dest;
498 core.data.read(&dest, HDF5::define_mem_type<Value_>(), core.memspace, core.dataspace);
499 fill(primary, dest);
500 } else {
501 fill(primary, 0);
502 }
503 return true;
504 } else {
505 return false;
506 }
507 }
