251 size_t copy_primary_to_buffer(IndexIt istart, DataIt dstart, size_t len, size_t first, size_t last, T* dbuffer, Thing thing) const {
252 size_t request_start = (first > *istart ? std::lower_bound(istart, istart + len, first) - istart : 0);
253 size_t request_end = (last < *(istart + len - 1) ? std::lower_bound(istart + request_start, istart + len, last) - istart : len);
254
255 if constexpr(std::is_same<typename std::remove_reference<Thing>::type, IDX*>::value) {
256 std::copy(istart + request_start, istart + request_end, thing);
257 std::copy(dstart + request_start, dstart + request_end, dbuffer);
258 } else {
259 std::fill(dbuffer, dbuffer + (last - first), 0);
260 for (size_t i = request_start; i < request_end; ++i) {
261 dbuffer[*(istart + i) - first] = *(dstart + i);
262 }
263 }
264
265 return request_end - request_start; // number of non-zeros.

366 template<typename Thing>
367 size_t extract_secondary(size_t i, T* dbuffer, Thing thing, size_t first, size_t last,
368 H5::DataSet& data, H5::DataSet& index,
369 H5::DataSpace& dataspace, H5::DataSpace& memspace,
370 std::vector<IDX>& index_cache) const
371 {
372 // Looping over all secondary dimensions.
373 size_t counter = 0;
374 for (size_t j = first; j < last; ++j) {
375 size_t left = pointers[j], right = pointers[j + 1];
376 index_cache.resize(right - left);
377
378 // Serial locks should be applied by the callers.
379 hsize_t offset = left;
380 hsize_t count = index_cache.size();
381 dataspace.selectHyperslab(H5S_SELECT_SET, &count, &offset);
382 memspace.setExtentSimple(1, &count);
383 memspace.selectAll();
384 index.read(index_cache.data(), HDF5::define_mem_type<IDX>(), memspace, dataspace);
385
386 auto it = std::lower_bound(index_cache.begin(), index_cache.end(), i);
387 if (it != index_cache.end() && *it == i) {
388 offset = left + (it - index_cache.begin());
389 count = 1;
390 dataspace.selectHyperslab(H5S_SELECT_SET, &count, &offset);
391 memspace.setExtentSimple(1, &count);
392 memspace.selectAll();
393
394 auto dest = dbuffer;
395 if constexpr(std::is_same<Thing, IDX*>::value) {
396 dest += counter;
397 } else {
398 dest += j - first;
399 }
400 data.read(dest, HDF5::define_mem_type<T>(), memspace, dataspace);
401
402 if constexpr(std::is_same<typename std::remove_reference<Thing>::type, IDX*>::value) {
403 thing[counter] = j;
404 ++counter;
405 }
406 }
407 }
408
409 return counter;
410 }
