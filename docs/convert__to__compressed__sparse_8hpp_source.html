26void count_compressed_sparse_non_zeros_consistent(const tatami::Matrix<Value_, Index_>* matrix, Index_ primary, Index_ secondary, bool row, Count_* output, int threads) {
27 if (matrix->is_sparse()) {
28 Options opt;
29 opt.sparse_extract_value = false;
30 opt.sparse_extract_index = false;
31 opt.sparse_ordered_index = false;
32
33 parallelize([&](int, Index_ start, Index_ length) -> void {
34 auto wrk = consecutive_extractor<true>(matrix, row, start, length, opt);
35 for (Index_ x = 0; x < length; ++x) {
36 auto range = wrk->fetch(NULL, NULL);
37 output[start + x] = range.number;
38 }
39 }, primary, threads);
40
41 } else {
42 parallelize([&](int, Index_ start, Index_ length) -> void {

61template<typename Value_, typename Index_, typename Count_>
62void count_compressed_sparse_non_zeros_inconsistent(const tatami::Matrix<Value_, Index_>* matrix, Index_ primary, Index_ secondary, bool row, Count_* output, int threads) {
63 std::vector<std::vector<Count_> > nz_counts(threads - 1);
64 for (auto& x : nz_counts) {
65 x.resize(primary);
66 }
67
68 if (matrix->is_sparse()) {
69 Options opt;
70 opt.sparse_extract_value = false;
71 opt.sparse_ordered_index = false;
72
73 parallelize([&](int t, Index_ start, Index_ length) -> void {
74 std::vector<Index_> buffer_i(primary);
75 auto wrk = consecutive_extractor<true>(matrix, !row, start, length, opt);
76 auto my_counts = (t > 0 ? nz_counts[t - 1].data() : output);
77
78 for (Index_ x = 0; x < length; ++x) {
79 auto range = wrk->fetch(NULL, buffer_i.data());
80#ifdef _OPENMP
81 #pragma omp simd
82#endif
83 for (Index_ i = 0; i < range.number; ++i) {
84 ++my_counts[range.index[i]];
85 }
86 }
87 }, secondary, threads);
88
89 } else {
90 parallelize([&](int t, Index_ start, Index_ length) -> void {
91 auto wrk = consecutive_extractor<false>(matrix, !row, start, length);
92 std::vector<Value_> buffer_v(primary);
93 auto my_counts = (t > 0 ? nz_counts[t - 1].data() : output);
94
95 for (Index_ x = 0; x < length; ++x) {
96 auto ptr = wrk->fetch(buffer_v.data());
97#ifdef _OPENMP
98 #pragma omp simd
99#endif
100 for (Index_ p = 0; p < primary; ++p) {
101 my_counts[p] += (ptr[p] != 0);
102 }
103 }
104 }, secondary, threads);
105 }
106
107 for (auto& y : nz_counts) {
108 for (Index_ p = 0; p < primary; ++p) {
109 output[p] += y[p];
110 }
111 }

114template<typename InputValue_, typename InputIndex_, typename Pointer_, typename StoredValue_, typename StoredIndex_>
115void fill_compressed_sparse_matrix_consistent(
116 const tatami::Matrix<InputValue_, InputIndex_>* matrix,
117 InputIndex_ primary,
118 InputIndex_ secondary,
119 bool row,
120 const Pointer_* pointers,
121 StoredValue_* output_value,
122 StoredIndex_* output_index,
123 int threads)
124{
125 if (matrix->is_sparse()) {
126 Options opt;
127 opt.sparse_ordered_index = false;
128
129 parallelize([&](int, InputIndex_ start, InputIndex_ length) -> void {
130 std::vector<InputValue_> buffer_v(secondary);
131 std::vector<InputIndex_> buffer_i(secondary);
132 auto wrk = consecutive_extractor<true>(matrix, row, start, length, opt);
133
134 for (InputIndex_ p = start, pe = start + length; p < pe; ++p) {
135 // Resist the urge to `fetch()` straight into 'output_v'
136 // and 'output_i', as implementations may assume that they
137 // have the entire 'length' length to play with, and the
138 // output vectors only have whatever is allocated from the
139 // first pass (which might be nothing for an all-zero matrix).
140 auto range = wrk->fetch(buffer_v.data(), buffer_i.data());
141 auto offset = pointers[p];
142 std::copy_n(range.value, range.number, output_value + offset);
143 std::copy_n(range.index, range.number, output_index + offset);
144 }
145 }, primary, threads);
146
147 } else {
148 parallelize([&](int, InputIndex_ start, InputIndex_ length) -> void {
149 std::vector<InputValue_> buffer_v(secondary);
150 auto wrk = consecutive_extractor<false>(matrix, row, start, length);
151
152 for (InputIndex_ p = start, pe = start + length; p < pe; ++p) {
153 auto ptr = wrk->fetch(buffer_v.data());
154 auto offset = pointers[p];
155 for (InputIndex_ s = 0; s < secondary; ++s) {
156 auto val = ptr[s];
157 if (val != 0) {
158 output_value[offset] = val;
159 output_index[offset] = s;
160 ++offset;
161 }
162 }
163 }
164 }, primary, threads);
165 }

346 auto& output_v = output.value;
347 auto& output_i = output.index;
348 auto& output_p = output.pointers;

350 InputIndex_ NR = matrix->nrow();
351 InputIndex_ NC = matrix->ncol();
352 InputIndex_ primary = (row ? NR : NC);
353 InputIndex_ secondary = (row ? NC : NR);
354
355 if (!two_pass) {
356 // Doing a single fragmented run and then concatenating everything together.
357 auto frag = retrieve_fragmented_sparse_contents<InputValue_, InputIndex_>(matrix, row, threads);
358 const auto& store_v = frag.value;
359 const auto& store_i = frag.index;
360
361 output_p.resize(static_cast<size_t>(primary) + 1);
362 for (InputIndex_ p = 0; p < primary; ++p) {
363 output_p[p + 1] = output_p[p] + store_v[p].size();
364 }
365
366 output_v.reserve(output_p.back());
367 output_i.reserve(output_p.back());
368 for (InputIndex_ p = 0; p < primary; ++p) {
369 output_v.insert(output_v.end(), store_v[p].begin(), store_v[p].end());
370 output_i.insert(output_i.end(), store_i[p].begin(), store_i[p].end());
371 }
