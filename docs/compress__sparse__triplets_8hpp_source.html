104 if constexpr(ROW) {

106 } else {
107 order_status = compress_triplets::is_ordered(cols, rows);
108 }
109
110 if (order_status != 0) {
111 std::vector<size_t> indices(N);
112 for (size_t i = 0; i < N; ++i) {
113 indices[i] = i;
114 }
115
116 // Sorting without duplicating the data.
117 if constexpr(ROW) {
118 compress_triplets::order(order_status, indices, rows, cols);
119 } else {
120 compress_triplets::order(order_status, indices, cols, rows);
121 }
122
123 // Reordering values in place. This (i) saves memory, and (ii) allows
124 // us to work with classes U and V that may not have well-defined copy
125 // constructors (e.g., if they refer to external memory).
126 for (size_t i = 0; i < indices.size(); ++i) {
127 if (indices[i] == static_cast<size_t>(-1)) {
128 continue;
129 }
130
131 size_t current = i, replacement = indices[i];
132 indices[i] = -1;
133
134 while (replacement != i) {
135 std::swap(rows[current], rows[replacement]);
136 std::swap(cols[current], cols[replacement]);
137 std::swap(values[current], values[replacement]);
138
139 current = replacement;
140 auto next_replacement = indices[replacement];
141 indices[replacement] = -1;
142 replacement = next_replacement;
143 }
144 }
145 }
146
147 // Collating the indices.
148 std::vector<size_t> output(ROW ? nr + 1 : nc + 1);
149 if constexpr(ROW) {
150 for (auto t : rows) {
151 ++(output[t+1]);
152 }
153 } else {
154 for (auto t : cols) {
155 ++(output[t+1]);
156 }
157 }
158 std::partial_sum(output.begin(), output.end(), output.begin());
