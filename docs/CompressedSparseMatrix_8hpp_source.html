134 void check_values(bool check) {
135 if (!check) {
136 return;
137 }
138
139 if (values.size() != indices.size()) {
140 throw std::runtime_error("'values' and 'indices' should be of the same length");
141 }
142
143 if (ROW) {
144 if (indptrs.size() != nrows + 1){
145 throw std::runtime_error("length of 'indptrs' should be equal to 'nrows + 1'");
146 }
147 } else {
148 if (indptrs.size() != ncols + 1){
149 throw std::runtime_error("length of 'indptrs' should be equal to 'ncols + 1'");
150 }
151 }
152
153 if (indptrs[0] != 0) {
154 throw std::runtime_error("first element of 'indptrs' should be zero");
155 }
156 if (indptrs[indptrs.size() - 1] != indices.size()) {
157 throw std::runtime_error("last element of 'indptrs' should be equal to length of 'indices'");
158 }
159
160 size_t counter = 0;
161 for (size_t i = 1; i < indptrs.size(); ++i) {
162 if (indptrs[i] < indptrs[i-1]) {
163 throw std::runtime_error("'indptrs' should be in increasing order");
164 }
165
166 if (counter < indices.size()) {
167 auto previous = indices[counter];
168 ++counter;
169 while (counter < indptrs[i]) {
170 if (previous >= indices[counter]) {
171 if (ROW) {
172 throw std::runtime_error("'indices' should be strictly increasing within each row");
173 } else {
174 throw std::runtime_error("'indices' should be strictly increasing within each column");
175 }
176 }
177 ++counter;
178 }
179 }
180 }
181
182 return;
183 }

186 std::pair<size_t, size_t> primary_dimension(size_t i, size_t first, size_t last, size_t otherdim) const {
187 const auto pstart = indptrs[i];
188 auto iIt = indices.begin() + pstart,
189 eIt = indices.begin() + indptrs[i + 1];
190 auto xIt = values.begin() + pstart;
191
192 if (first) { // Jumping ahead if non-zero.
193 iIt = std::lower_bound(iIt, eIt, first);
194 }
195
196 if (last != otherdim) { // Jumping to last element.
197 eIt = std::lower_bound(iIt, eIt, last);
198 }
199
200 return std::make_pair(iIt - indices.begin(), eIt - iIt);

252 struct CompressedSparseWorkspace : public Workspace {
253 CompressedSparseWorkspace(size_t max_index, const V& idx, const W& idp) :
254 previous_request(idp.size() - 1),
255 current_indptrs(idp.begin(), idp.begin() + idp.size() - 1), // all but the last.
256 current_indices(idp.size() - 1)
257 {
258 /* Here, the general idea is to store a local copy of the actual
259 * row indices (for CSC matrices; column indices, for CSR matrices)
260 * so that we don't have to keep on doing cache-unfriendly look-ups
261 * for the indices based on the pointers that we do have. This assumes
262 * that the density is so low that updates to the local indices are
263 * rare relative to the number of comparisons to those same indices.
264 * Check out the `secondary_dimension()` function for how this is used.
265 */
266 for (size_t i = 0; i < idp.size() - 1; ++i) {
267 current_indices[i] = (idp[i] < idp[i+1] ? idx[idp[i]] : max_index);
268 }
269 return;
270 }
271
272 std::vector<size_t> previous_request; // the last request for each column.
273 std::vector<typename std::remove_reference<decltype(std::declval<W>()[0])>::type> current_indptrs; // the current position of the pointer
274 std::vector<typename std::remove_reference<decltype(std::declval<V>()[0])>::type> current_indices; // the current index being pointed to
275 };

287 void secondary_dimension(IDX i, size_t first, size_t last, Workspace* work, STORE& output) const {
288 if (work == nullptr) {
289 for (size_t c = first; c < last; ++c) {
290 auto start = indices.begin() + indptrs[c];
291 auto end = indices.begin() + indptrs[c+1];
292 auto iIt = std::lower_bound(start, end, i);
293 if (iIt != end && *iIt == i) {
294 auto offset = iIt - indices.begin();
295 output.add(c, values[offset]);
296 }
297 }
298 } else {
299 CompressedSparseWorkspace& worker = *(dynamic_cast<CompressedSparseWorkspace*>(work));
300 auto max_index = max_secondary_index();
301
302 for (size_t current = first; current < last; ++current) {
303 auto& prev_i = worker.previous_request[current];
304 auto& curptr = worker.current_indptrs[current];
305 auto& curdex = worker.current_indices[current];
306
307 if (i == prev_i + 1) {
308 if (curdex < i) { // if true, this implies that curptr < indptrs[current + 1], provided i < max_index.
309 ++curptr;
310 curdex = (curptr < indptrs[current + 1] ? indices[curptr] : max_index);
311 }
312 } else if (i + 1 == prev_i) {
313 if (curptr != indptrs[current] && indices[curptr - 1] >= i) {
314 --curptr;
315 curdex = indices[curptr];
316 }
317 } else if (i > prev_i) {
318 if (curdex < i) { // same implication as above.
319 curptr = std::lower_bound(indices.begin() + curptr, indices.begin() + indptrs[current + 1], i) - indices.begin();
320 curdex = (curptr < indptrs[current + 1] ? indices[curptr] : max_index);
321 }
322 } else if (i < prev_i) {
323 if (curptr != indptrs[current]) {
324 curptr = std::lower_bound(indices.begin() + indptrs[current], indices.begin() + curptr, i) - indices.begin();
325 curdex = indices[curptr];
326 }
327 }
328
329 prev_i = i;
330 if (curdex == i) { // assuming i < max_index, of course.
331 output.add(current, values[curptr]);
332 }
333 }
334 }
335 return;
336 }
