89 throw std::runtime_error("first element of 'indptrs' should be zero");

92 throw std::runtime_error("last element of 'indptrs' should be equal to length of 'indices'");

95 size_t counter = 0;

152 SecondaryWorkspaceBase(size_t max_index, const V& idx, const W& idp, size_t start, size_t length) :
153 current_indptrs(idp.begin() + start, idp.begin() + start + length), current_indices(length)
154 {
155 /* Here, the general idea is to store a local copy of the actual
156 * row indices (for CSC matrices; column indices, for CSR matrices)
157 * so that we don't have to keep on doing cache-unfriendly look-ups
158 * for the indices based on the pointers that we do have. This assumes
159 * that the density is so low that updates to the local indices are
160 * rare relative to the number of comparisons to those same indices.
161 * Check out the `secondary_dimension()` function for how this is used.
162 */
163 auto idpIt = idp.begin() + start;
164 for (size_t i = 0; i < length; ++i, ++idpIt) {
165 current_indices[i] = (*idpIt < *(idpIt + 1) ? idx[*idpIt] : max_index);
166 }
167 return;
168 }

173 SecondaryWorkspaceBase(size_t max_index, const V& idx, const W& idp, const std::vector<IDX>& subset) :
174 current_indptrs(subset.size()), current_indices(subset.size())
175 {

177 auto i = subset[i0];
178 current_indptrs[i0] = idp[i];
179 current_indices[i0] = (idp[i] < idp[i + 1] ? idx[idp[i]] : max_index);
180 }

434 std::pair<size_t, size_t> primary_dimension(size_t i, size_t start, size_t length, size_t otherdim, PrimaryBlockWorkspaceBase* work) const {
435 bool do_cache = work && !work->cached.empty();
436 if (do_cache) {
437 auto val = work->cached[i];
438 if (val.first != -1) {
439 return val;
440 }
441 }
442
443 auto iIt = indices.begin() + indptrs[i], eIt = indices.begin() + indptrs[i + 1];
444
445 if (start) { // Jumping ahead if non-zero.
446 iIt = std::lower_bound(iIt, eIt, start);
447 }
448
449 auto last = start + length;
450 if (last != otherdim) { // Jumping to last element.
451 eIt = std::lower_bound(iIt, eIt, last);
452 }
453
454 size_t outstart = iIt - indices.begin();
455 size_t outlength = eIt - iIt;
456 if (do_cache) {
457 work->cached[i].first = outstart;
458 work->cached[i].second = outlength;
459 }
460
461 return std::make_pair(outstart, outlength);
462 }

515 auto define_below_index(size_t pos, size_t curptr) const {
516 // We set it to zero if there is no other non-zero element before
517 // 'curptr' in 'pos'. This aligns with the idea of 'below_indices'
518 // holding 1-past-the-immediately-below-index; if it is zero,
519 // nothing remaings below 'curptr'.
520 return (curptr > indptrs[pos] ? indices[curptr - 1] + 1 : 0);
521 }

535 template<class Store>
536 void secondary_dimension_above(IDX secondary, size_t primary, size_t index_primary, SecondaryWorkspaceBase& work, Store& output) const {
537 auto& curdex = work.current_indices[index_primary];
538 auto& curptr = work.current_indptrs[index_primary];
539
540 // Remember that we already store the index corresponding to the current indptr.
541 // So, we only need to do more work if the request is greater than that index.
542 if (secondary > curdex) {
543 auto max_index = max_secondary_index();
544 auto limit = indptrs[primary + 1];
545
546 // Having a peek at the index of the next non-zero
547 // element; maybe we're lucky enough that the requested
548 // index is below this, as would be the case for
549 // consecutive or near-consecutive accesses.
550 ++curptr;
551 if (curptr < limit) {
552 auto candidate = indices[curptr];
553 if (candidate >= secondary) {
554 curdex = candidate;
555
556 } else if (secondary + 1 == max_index) {
557 // Special case if the requested index is at the end of the matrix,
558 // in which case we can just jump there directly rather than
559 // doing an unnecessary binary search.
560 curptr = limit - 1;
561 if (indices[curptr] == secondary) {
562 curdex = indices[curptr];
563 } else {
564 ++curptr;
565 curdex = max_index;
566 }
567
568 } else {
569 // Otherwise we need to search.
570 curptr = std::lower_bound(indices.begin() + curptr, indices.begin() + limit, secondary) - indices.begin();
571 curdex = (curptr < limit ? indices[curptr] : max_index);
572 }
573 } else {
574 curdex = max_index;
575 }
576
577 if (work.store_below) {
578 work.below_indices[index_primary] = define_below_index(primary, curptr);
579 }
580 }
581
582 if (secondary == curdex) { // assuming secondary < max_index, of course.
583 add_to_store(primary, curptr, output);
584 } else {
585 output.skip(primary);
586 }
587 }

589 template<class Store>
590 void secondary_dimension_below(IDX secondary, size_t primary, size_t index_primary, SecondaryWorkspaceBase& work, Store& output) const {
591 auto& curdex = work.current_indices[index_primary];
592 auto& curptr = work.current_indptrs[index_primary];
593
594 // Remember that below_indices stores 'indices[curptr - 1] + 1'.
595 // So, we only need to do more work if the request is less than this;
596 // otherwise the curptr remains unchanged. We also skip if curptr is
597 // equal to 'indptrs[index_primary]' because decreasing is impossible.
598 auto& prevdex = work.below_indices[index_primary];
599 auto limit = indptrs[primary];
600 if (secondary < prevdex && curptr > limit) {
601
602 // Having a peek at the index of the non-zero element below us.
603 // Maybe we're lucky and the requested index is equal to this,
604 // in which case we can use it directly. This would be the case
605 // for consecutive accesses in the opposite direction.
606 auto candidate = indices[curptr - 1];
607 if (candidate == secondary) {
608 --curptr;
609 curdex = candidate;
610
611 } else if (secondary == 0) {
612 // Special case if the requested index is at the start of the matrix,
613 // in which case we can just jump there directly rather than
614 // doing an unnecessary binary search.
615 curdex = indices[limit];
616 curptr = limit;
617
618 } else {
619 // Otherwise we need to search. Note that there's no need to
620 // handle candidate < secondary, as that would imply that
621 // 'secondary >= prevdex', which isn't possible.
622 curptr = std::lower_bound(indices.begin() + limit, indices.begin() + curptr - 1, secondary) - indices.begin();
623 curdex = indices[curptr]; // guaranteed to be valid as curptr - 1 >= limit.
624 }
625
626 prevdex = define_below_index(primary, curptr);
627 }
628
629 if (secondary == curdex) { // assuming i < max_index, of course.
630 add_to_store(primary, curptr, output);
631 } else {
632 output.skip(primary);
633 }
634 }

636 template<class STORE>
637 void secondary_dimension(IDX i, size_t start, size_t length, SecondaryWorkspaceBase& work, STORE& output) const {
638 IDX prev_i = work.previous_request;
639
640 if (i >= prev_i) {
641 for (size_t current = 0; current < length; ++current) {
642 secondary_dimension_above(i, current + start, current, work, output);
643 }
644 } else {
645 // Backfilling everything so that all uses of 'below_indices' are valid.
646 if (!work.store_below) {
647 work.store_below = true;
648 work.below_indices.resize(length);
649 for (size_t j = 0; j < length; ++j) {
650 work.below_indices[j] = define_below_index(j + start, work.current_indptrs[j]);
651 }
652 }
653
654 for (size_t current = 0; current < length; ++current) {

663 // Overview of the Store contract:
664 struct raw_store {
665 T* out_values;
666 IDX* out_indices;
667 size_t n = 0;
668
669 // Whether or not 'values' might be ignored, typically
670 // because a NULL pointer is supplied in the 'vbuffer'
671 static constexpr bool can_ignore_values = true;
672
673 // If 'can_ignore_values = true', we need this method
674 // to define whether the values are ignored for this instance.
675 bool ignore_values() const {
676 return !out_values;
677 }
678
679 // If 'can_ignore_values = true', we need this method to handle
680 // hits, but _without_ a costly fetch of the value from memory.
681 void add(IDX i) {
682 ++n;
683 if (out_indices) {
684 *out_indices = i;
685 ++out_indices;
686 }
687 }
688
689 // Storing a hit in terms of its index and its value. If 'can_ignore_values = true',
690 // it can be assumed that 'skip_values()' is false if this method is called. Note
691 // that add() and skip() is always called on indices in increasing order.
692 void add(IDX i, T val) {
693 add(i);
694 *out_values = val;
695 ++out_values;
696 return;
697 }
698
699 // Skip an index if it isn't found in the data.
700 void skip(IDX) {}
701 };

849 template<class Store>
850 void primary_dimension(size_t i, const std::vector<IDX>& subset, PrimaryIndexedWorkspaceBase& work, Store& store) const {
851 auto iIt = indices.begin() + indptrs[i], eIt = indices.begin() + indptrs[i + 1];
852
853 if (!subset.empty() && subset[0]) { // Jumping ahead if non-zero.
854 bool do_cache = !work.starts.empty();
855 if (do_cache) {
856 if (work.starts[i] != -1) { // retrieving the jump from cache, if we came here before.
857 iIt += work.starts[i];
858 } else {
859 auto iIt2 = std::lower_bound(iIt, eIt, subset[0]);
860 work.starts[i] = iIt2 - iIt;
861 iIt = iIt2;
862 }
863 } else {
864 iIt = std::lower_bound(iIt, eIt, subset[0]);
865 }
866 }
867
868 if (iIt == eIt) {
869 return;
870 }
871
872 size_t counter = 0, end = subset.size();
873 while (counter < end) {
874 auto current = subset[counter];
875
876 while (iIt != eIt && current > *iIt) {
877 ++iIt;
878 }
879 if (iIt == eIt) {
880 break;
881 }
882
883 if (current == *iIt) {
884 add_to_store(current, iIt - indices.begin(), store);
885 } else {
886 store.skip(current);
887 }
888 ++counter;
889 }
890
891 return;
892 }
