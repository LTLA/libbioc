396public:
411 CompressedSparseMatrix(Index_ nr, Index_ nc, ValueStorage_ vals, IndexStorage_ idx, PointerStorage_ ptr, bool check=true) :
412 nrows(nr), ncols(nc), values(std::move(vals)), indices(std::move(idx)), indptrs(std::move(ptr))
413 {
414 if (check) {
415 if (values.size() != indices.size()) {
416 throw std::runtime_error("'values' and 'indices' should be of the same length");
417 }
418
419 if constexpr(row_) {
420 if (indptrs.size() != static_cast<size_t>(nrows) + 1){
421 throw std::runtime_error("length of 'indptrs' should be equal to 'nrows + 1'");
422 }
423 } else {
424 if (indptrs.size() != static_cast<size_t>(ncols) + 1){
425 throw std::runtime_error("length of 'indptrs' should be equal to 'ncols + 1'");
426 }
427 }
428
429 if (indptrs[0] != 0) {
430 throw std::runtime_error("first element of 'indptrs' should be zero");
431 }
432
433 auto last = indptrs[indptrs.size() - 1]; // don't use back() as this is not guaranteed to be available for arbitrary PointerStorage_.
434 if (static_cast<size_t>(last) != indices.size()) {
435 throw std::runtime_error("last element of 'indptrs' should be equal to length of 'indices'");
436 }
437
438 ElementType<IndexStorage_> max_index = (row_ ? ncols : nrows);
439 for (size_t i = 1; i < indptrs.size(); ++i) {
440 auto start = indptrs[i- 1], end = indptrs[i];
441 if (end < start || end > last) {
442 throw std::runtime_error("'indptrs' should be in non-decreasing order");
443 }
444
445 for (auto x = start; x < end; ++x) {
446 if (indices[x] < 0 || indices[x] >= max_index) {
447 if constexpr(row_) {
448 throw std::runtime_error("'indices' should contain non-negative integers less than the number of rows");
449 } else {
450 throw std::runtime_error("'indices' should contain non-negative integers less than the number of columns");
451 }
452 }
453 }
454
455 for (size_t j = start + 1; j < end; ++j) {
456 if (indices[j] <= indices[j - 1]) {
457 if constexpr(row_) {
458 throw std::runtime_error("'indices' should be strictly increasing within each row");
459 } else {
460 throw std::runtime_error("'indices' should be strictly increasing within each column");
461 }
462 }
463 }
464 }
465 }
466 }
