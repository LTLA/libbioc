77 void check_values(bool check) {
78 if (!check) {
79 return;
80 }
81
82 if (values.size() != indices.size()) {
83 throw std::runtime_error("'values' and 'indices' should be of the same length");
84 }
85
86 if (row_) {
87 if (indptrs.size() != static_cast<size_t>(nrows) + 1){
88 throw std::runtime_error("length of 'indptrs' should be equal to 'nrows + 1'");
89 }
90 } else {
91 if (indptrs.size() != static_cast<size_t>(ncols) + 1){
92 throw std::runtime_error("length of 'indptrs' should be equal to 'ncols + 1'");
93 }
94 }
95
96 if (indptrs[0] != 0) {
97 throw std::runtime_error("first element of 'indptrs' should be zero");
98 }
99 if (static_cast<size_t>(indptrs[indptrs.size() - 1]) != indices.size()) {
100 throw std::runtime_error("last element of 'indptrs' should be equal to length of 'indices'");
101 }
102
103 size_t counter = 0;
104 for (size_t i = 1; i < indptrs.size(); ++i) {
105 if (indptrs[i] < indptrs[i-1]) {
106 throw std::runtime_error("'indptrs' should be in increasing order");
107 }
108
109 if (counter < indices.size()) {
110 auto previous = indices[counter];
111 ++counter;
112 while (counter < static_cast<size_t>(indptrs[i])) {
113 if (previous >= indices[counter]) {
114 if (row_) {
115 throw std::runtime_error("'indices' should be strictly increasing within each row");
116 } else {
117 throw std::runtime_error("'indices' should be strictly increasing within each column");
118 }
119 }
120 ++counter;
121 }
122 }
123 }
124
125 return;
126 }

213 std::pair<size_t, size_t> primary_dimension(Index_ i, Index_ start, Index_ length, PrimaryWorkspace& work) const {
214 bool do_cache = !(work.cached.empty());
215 if (do_cache) {
216 auto val = work.cached[i];
217 if (val.first != -1) {
218 return val;
219 }
220 }
221
222 auto iIt = indices.begin() + indptrs[i], eIt = indices.begin() + indptrs[i + 1];
223
224 if (start) { // Jumping ahead if non-zero.
225 iIt = std::lower_bound(iIt, eIt, start);
226 }
227
228 auto last = start + length;
229 if (last != max_secondary_index()) { // Jumping to last element.
230 eIt = std::lower_bound(iIt, eIt, last);
231 }
232
233 size_t outstart = iIt - indices.begin();
234 size_t outlength = eIt - iIt;
235 if (do_cache) {
236 work.cached[i].first = outstart;
237 work.cached[i].second = outlength;
238 }
239
240 return std::make_pair(outstart, outlength);
241 }

287 void add_to_store(Index_ primary, size_t curptr, Store_& output) const {
288 // If the value isn't actually needed, we don't want to do a look-up on 'values'.
289 if constexpr(Store_::can_ignore_values) {
290 if (output.ignore_values()) {
291 output.add(primary);
292 return;
293 }

298 template<class Store_>
299 void primary_dimension(Index_ i, const Index_* subset, Index_ length, PrimaryWorkspace& work, Store_& store) const {
300 if (!length) {
301 return;
302 }
303
304 auto iIt = indices.begin() + indptrs[i], eIt = indices.begin() + indptrs[i + 1];
305
306 if (indices[0]) { // Only jumping ahead if the start is non-zero.
307 bool do_cache = !work.cached.empty();
308 if (do_cache) {
309 if (work.cached[i].first != -1) { // retrieving the jump from cache, if we came here before.
310 iIt += work.cached[i].first;
311 } else {
312 auto iIt2 = std::lower_bound(iIt, eIt, subset[0]);
313 work.cached[i].first = iIt2 - iIt;
314 iIt = iIt2;
315 }
316 } else {
317 iIt = std::lower_bound(iIt, eIt, subset[0]);
318 }
319 }
320
321 if (iIt == eIt) {
322 return;
323 }
324
325 Index_ counter = 0;
326 while (counter < length) {
327 auto current = subset[counter];
328
329 while (iIt != eIt && current > *iIt) {
330 ++iIt;
331 }
332 if (iIt == eIt) {
333 break;
334 }
335
336 if (current == *iIt) {
337 add_to_store(current, iIt - indices.begin(), store);
338 } else {
339 store.skip(current);
340 }
341 ++counter;
342 }
343
344 return;
345 }

347 // Overview of the Store contract:
348 struct RawStore {
349 Value_* out_values;
350 Index_* out_indices;
351 Index_ n = 0;
352
353 // Whether or not 'values' might be ignored, typically
354 // because a NULL pointer is supplied in the 'vbuffer'
355 static constexpr bool can_ignore_values = true;
356
357 // If 'can_ignore_values = true', we need this method
358 // to define whether the values are ignored for this instance.
359 bool ignore_values() const {
360 return !out_values;
361 }
362
363 // If 'can_ignore_values = true', we need this method to handle
364 // hits, but _without_ a costly fetch of the value from memory.
365 void add(Index_ i) {
366 ++n;
367 if (out_indices) {
368 *out_indices = i;
369 ++out_indices;
370 }
371 }
372
373 // Storing a hit in terms of its index and its value. If 'can_ignore_values = true',
374 // it can be assumed that 'skip_values()' is false if this method is called. Note
375 // that add() and skip() is always called on indices in increasing order.
376 void add(Index_ i, Value_ val) {
377 add(i);
378 *out_values = val;
379 ++out_values;
380 return;
381 }
382
383 // Skip an index if it isn't found in the data.
384 void skip(Index_) {}
385 };
