10/*
11 * If PointerStorage_ is just a bool, then we assume that the indices for each
12 * primary element are fragmented into separate vectors. In such cases, we can
13 * just call size() to get the upper limit and we set the lower limit to zero.
14 *
15 * If PointerStorage_ is not a bool, it is assumed to be a vector of pointers
16 * in the typical compressed sparse style. In such cases, we actually need to
17 * do some work to get the lower/upper bounds of each primary element.
18 */
19
20template<class PointerStorage_>
21constexpr bool is_fragmented() {
22 return std::is_same<PointerStorage_, bool>::value;
23}
24
25template<class PointerStorage_, class IndexStorage_, typename Index_>
26const auto& get_indices(const IndexStorage_& all_indices, Index_ primary) {
27 if constexpr(is_fragmented<PointerStorage_>()) {
28 return all_indices[primary];
29 } else {
30 return all_indices;
31 }
32}
33
34template<class IndexStorage_, class PointerStorage_, typename Index_>
35auto get_upper_limit(const IndexStorage_& indices, const PointerStorage_& indptrs, Index_ primary) {
36 if constexpr(is_fragmented<PointerStorage_>()) {
37 static_assert(std::is_arithmetic<decltype(indices[0])>::value); // just making sure we got passed in the output of 'get_indices'.
38 return indices.size();
39 } else {
40 return indptrs[primary + 1];
41 }
42}
43
44template<class PointerStorage_, typename Index_>
45auto get_lower_limit(const PointerStorage_& indptrs, Index_ primary) {
46 if constexpr(is_fragmented<PointerStorage_>()) {
47 return 0;
48 } else {
49 return indptrs[primary];
50 }
51}
