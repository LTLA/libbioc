67 template<class Store>
68 void new_line(Store& store) {
69 if (in_comment) {
70 in_comment = false;
71 } else {
72 if ((onto == 3 && !non_empty) || (onto == 2 && non_empty)) {
73 // i.e., there are three fields.
74 ;
75 } else {
76 throw std::runtime_error("line " + std::to_string(current_line + 1) + " should contain three values");
77 }
78
79 if (!passed_preamble) {
80 nrows = currow;
81 ncols = curcol;
82 nlines = curval;
83
84 store.setdim(currow, curcol, curval);
85 passed_preamble = true;
86 } else {

89 }

92 }

96 }

99 }

102 throw std::runtime_error("more lines present than specified in the header (" + std::to_string(nlines) + ")");
103 }
104
105 if constexpr(!preamble_only<Store>::value) {
106 store.addline(currow - 1, curcol - 1, curval, current_data_line);
107 }
108 ++current_data_line;
109 }
110
111 onto = 0;
112 currow = 0;
113 curcol = 0;
114 curval = 0;
115 non_empty = false;
116 }

118 }

121 template<class Reader, class Store>
122 void operator()(Reader& reader, Store& store) {
123 bool remaining;
124
125 do {
126 remaining = reader();
127 auto buffer = reinterpret_cast<const char*>(reader.buffer());
128 auto n = reader.available();
129
130 size_t i = 0;
131 while (i < n) {
132 if (buffer[i] == '%') {
133 in_comment = true;
134
135 // Try to get quickly to the next line.
136 do {
137 ++i;
138 } while (i < n && buffer[i] != '\n');
139 } else if (buffer[i] == '\n') {
140 new_line(store);
141 ++i;
142
143 // Breaking out early if we only care about the preamble.
144 if constexpr(preamble_only<Store>::value) {
145 if (passed_preamble) {
146 return;
147 }
148 }
149 } else if (in_comment) {
150 // Try to get to the next line, again.
151 do {
152 ++i;
153 } while (i < n && buffer[i] != '\n');
154 } else {
155 while (i < n) {
156 if (!std::isdigit(buffer[i])) {
157 if (!std::isspace(buffer[i])) {
158 throw std::runtime_error("values should be non-negative integers on line " + std::to_string(current_line));
159 }
160
161 if (buffer[i] == '\n') {
162 break;
163 }
164
165 if (non_empty) {
166 ++onto;
167 non_empty = false;
168 }
169 ++i;
170 continue;
171 }
172
173 non_empty = true; // check that something was _actually_ present in this field.
174 int delta = buffer[i] - '0';
175 switch(onto) {
176 case 0:
177 currow *= 10;
178 currow += delta;
179 break;
180 case 1:
181 curcol *= 10;
182 curcol += delta;
183 break;
184 case 2:
185 curval *= 10;
186 curval += delta;
187 break;
188 }
189 ++i;
190 }
191 }
192 }
193 } while (remaining);
194
195 // If onto = 0 and non_empty = false, we ended on a newline, so
196 // there's no extra entry to add. Otherwise, we try to add the
197 // last line that was _not_ terminated by a newline.
198 if (onto != 0 || non_empty) {
199 new_line(store);
200 }
201
202 if (!passed_preamble) {
203 throw std::runtime_error("no header line specifying the dimensions");
204 }
205
206 if (current_data_line != nlines) {
207 throw std::runtime_error("detected " + std::to_string(current_data_line) + " lines but " + std::to_string(nlines) + " lines specified in the header");
208 }
209 return;
210 }
