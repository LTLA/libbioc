20template<ArithmeticOperation op_, bool right_, typename InputValue_, typename Index_, typename Scalar_, typename OutputValue_>
21void delayed_arithmetic_run_simple(const InputValue_* input, Index_ length, Scalar_ scalar, OutputValue_* output) {
22 for (Index_ i = 0; i < length; ++i) {
23 if constexpr(std::is_same<InputValue_, OutputValue_>::value) {
24 auto& val = output[i];
25 val = delayed_arithmetic<op_, right_>(val, scalar);
26 } else {
27 output[i] = delayed_arithmetic<op_, right_>(input[i], scalar);
28 }
29 }
30}

32// Previously, we applied the operation empirically and checked whether the
33// result is zero. This was not ideal as some operations may yield undefined
34// values for certain input types (e.g., division by zero involving integers),
35// which could cause compile-time errors. It also requires our helper class to
36// know the input matrix type ahead of time, which should not be necessary.
37template<ArithmeticOperation op_, bool right_, typename Scalar_>
38bool delayed_arithmetic_actual_sparse(Scalar_ scalar) {
39 if constexpr(op_ == ArithmeticOperation::ADD || op_ == ArithmeticOperation::SUBTRACT) {
40 return (scalar == 0);
41 } else if constexpr(op_ == ArithmeticOperation::MULTIPLY) {
42 // Need to account for 'scalar' being Inf or NaN, in which case it is not sparse.
43 return (0 * scalar == 0);
44 } else if constexpr(op_ == ArithmeticOperation::POWER) {
45 return (right_ && scalar > 0);
46 } else { // DIVIDE, MODULO and INTEGER_DIVIDE
47 return (right_ && (scalar > 0 || scalar < 0)); // don't use '!=' as this would be true for NaNs.
48 }
49}

51template<ArithmeticOperation op_, bool right_, typename OutputValue_, typename Scalar_>
52OutputValue_ delayed_arithmetic_zero(Scalar_ scalar) {
53 if constexpr(has_unsafe_divide_by_zero<op_, OutputValue_, Scalar_>()) {
54 if constexpr(right_) {
55 if (scalar) {
56 return delayed_arithmetic<op_, right_, OutputValue_>(0, scalar);
57 }
58 }
59 throw std::runtime_error("division by zero is not supported");
60 return 0;
61 } else {
62 return delayed_arithmetic<op_, right_, OutputValue_>(0, scalar);
63 }
