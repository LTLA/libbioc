49CompressedSparseMatrix<ROW, T, IDX, U, V, W> load_hdf5_compressed_sparse_matrix(
50 size_t nr,
51 size_t nc,
52 const std::string& file,
53 const std::string& vals,
54 const std::string& idx,
55 const std::string& ptr)
56{
57 H5::H5File file_handle(file, H5F_ACC_RDONLY);
58
59 auto dhandle = HDF5::open_and_check_dataset<false>(file_handle, vals);
60 const size_t nonzeros = HDF5::get_array_dimensions<1>(dhandle, "vals")[0];
61 U x(nonzeros);
62 dhandle.read(x.data(), HDF5::define_mem_type<typename U::value_type>());
63
64 auto ihandle = HDF5::open_and_check_dataset<true>(file_handle, idx);
65 if (HDF5::get_array_dimensions<1>(ihandle, "idx")[0] != nonzeros) {
66 throw std::runtime_error("number of non-zero elements is not consistent between 'data' and 'idx'");
67 }
68 V i(nonzeros);
69 ihandle.read(i.data(), HDF5::define_mem_type<typename V::value_type>());
70
71 auto phandle = HDF5::open_and_check_dataset<true>(file_handle, ptr);
72 const size_t ptr_size = HDF5::get_array_dimensions<1>(phandle, "ptr")[0];
73 if (ptr_size != (ROW ? nr : nc) + 1) {
74 throw std::runtime_error("'ptr' dataset should have length equal to the number of " + (ROW ? std::string("rows") : std::string("columns")) + " plus 1");
75 }
76
77 // Because HDF5 doesn't have a native type for size_t.
78 W p(ptr_size);
79 if constexpr(std::is_same<size_t, typename W::value_type>::value) {
80 if constexpr(std::is_same<size_t, hsize_t>::value) {
81 phandle.read(p.data(), H5::PredType::NATIVE_HSIZE);
82 } else {
83 std::vector<hsize_t> p0(ptr_size);
84 phandle.read(p0.data(), H5::PredType::NATIVE_HSIZE);
85 std::copy(p0.begin(), p0.end(), p.begin());
86 }
87 } else {
88 phandle.read(p.data(), HDF5::define_mem_type<typename W::value_type>());
89 }
90
91 return CompressedSparseMatrix<ROW, T, IDX, U, V, W>(nr, nc, std::move(x), std::move(i), std::move(p));
92}
