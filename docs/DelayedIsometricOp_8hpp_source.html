130 template<bool accrow_, DimensionSelectionType selection_, class IsometricExtractor_>
131 void mutate_dense(const Value_* ptr, Value_* buffer, Index_ i, const IsometricExtractor_* exptr) const {
132 if constexpr(selection_ == DimensionSelectionType::FULL) {
133 auto end = exptr->full_length;
134 for (Index_ j = 0; j < end; ++j, ++ptr) {
135 if constexpr(accrow_) {
136 buffer[j] = operation(i, j, *ptr);
137 } else {
138 buffer[j] = operation(j, i, *ptr);
139 }
140 }
141
142 } else if constexpr(selection_ == DimensionSelectionType::BLOCK) {
143 auto end = exptr->block_length;
144 auto shift = exptr->block_start;
145 for (Index_ j = 0; j < end; ++j, ++ptr) {
146 if constexpr(accrow_) {
147 buffer[j] = operation(i, j + shift, *ptr);
148 } else {
149 buffer[j] = operation(j + shift, i, *ptr);
150 }
151 }
152
153 } else {
154 auto xptr = exptr->internal->index_start();
155 auto end = exptr->index_length;
156 for (Index_ j = 0; j < end; ++j, ++ptr, ++xptr) {
157 if constexpr(accrow_) {
158 buffer[j] = operation(i, *xptr, *ptr);
159 } else {
160 buffer[j] = operation(*xptr, i, *ptr);
161 }
162 }
163 }
164 }

167 ********** Sparse extraction **********
168 ***************************************/
169private:
170 // This extractor doesn't actually require the indices on the extraction
171 // dimension e.g., because the operator doesn't rely on position. So we
172 // don't have to guarantee the extraction of the indices if the user didn't
173 // provide space in the 'ibuffer'.
174 template<bool accrow_, DimensionSelectionType selection_>
175 struct SimpleSparseIsometricExtractor : public IsometricExtractorBase<selection_, true, true> {
176 SimpleSparseIsometricExtractor(const DelayedIsometricOp* p, std::unique_ptr<Extractor<selection_, true, Value_, Index_> > i) :
177 IsometricExtractorBase<selection_, true, true>(p, std::move(i)) {}
178
179 SparseRange<Value_, Index_> fetch(Index_ i, Value_* vbuffer, Index_* ibuffer) {
180 auto raw = this->internal->fetch(i, vbuffer, ibuffer);
181
182 if (raw.value) {
183 const auto& OP = this->parent->operation;
184 for (Index_ j = 0; j < raw.number; ++j) {
185 if constexpr(accrow_) {
186 vbuffer[j] = OP(i, 0, raw.value[j]); // no-op value, we don't need the column indices.
187 } else {
188 vbuffer[j] = OP(0, i, raw.value[j]); // no-op value, we don't need the row indices.
189 }
190 }
191 raw.value = vbuffer;
192 }
193
194 return raw;
195 }

198 template<bool accrow_, DimensionSelectionType selection_>
199 struct RegularSparseIsometricExtractor : public IsometricExtractorBase<selection_, true, true> {
200 RegularSparseIsometricExtractor(const DelayedIsometricOp* p, std::unique_ptr<Extractor<selection_, true, Value_, Index_> > i, bool ri, bool report_value) :
201 IsometricExtractorBase<selection_, true, true>(p, std::move(i)), report_index(ri)
202 {
203 if (!report_index && report_value) {
204 // We only need an internal ibuffer if the user wants the
205 // values but didn't provide enough space to store the indices
206 // (which we need to pass to the operation's functor).
207 internal_ibuffer.resize(extracted_length<selection_, Index_>(*this));
208 }
209 }
210
211 SparseRange<Value_, Index_> fetch(Index_ i, Value_* vbuffer, Index_* ibuffer) {
212 // If the internal ibuffer is empty, we're either extracting the indices
213 // directly into the user's ibuffer, or we don't need the indices at all.
214 // Either way, it doesn't hurt to use the user's ibuffer.
215 Index_* iin = (internal_ibuffer.empty() ? ibuffer : internal_ibuffer.data());
216
217 auto raw = this->internal->fetch(i, vbuffer, iin);
218
219 if (raw.value) {
220 const auto& OP = this->parent->operation;
221 for (Index_ j = 0; j < raw.number; ++j) {
222 if constexpr(accrow_) {
223 vbuffer[j] = OP(i, raw.index[j], raw.value[j]);
224 } else {
225 vbuffer[j] = OP(raw.index[j], i, raw.value[j]);
226 }
227 }
228 raw.value = vbuffer;
229 }
230
231 if (!report_index) {
232 raw.index = NULL;
233 }
234
235 return raw;
236 }
237
238 protected:
239 std::vector<Index_> internal_ibuffer;
240 bool report_index = false;
241 };

250 template<bool accrow_, DimensionSelectionType selection_>
251 struct DensifiedSparseIsometricExtractor : public IsometricExtractorBase<selection_, true, false> {
252 DensifiedSparseIsometricExtractor(const DelayedIsometricOp* p, std::unique_ptr<Extractor<selection_, false, Value_, Index_> > i, bool ri, bool rv) :
253 IsometricExtractorBase<selection_, true, false>(p, std::move(i)), report_index(ri), report_value(rv)
254 {}
255
256 SparseRange<Value_, Index_> fetch(Index_ i, Value_* vbuffer, Index_* ibuffer) {
257 SparseRange<Value_, Index_> raw(extracted_length<selection_, Index_>(*this), NULL, NULL);
258
259 if (report_value) {
260 auto ptr = this->internal->fetch(i, vbuffer);
261 this->parent->template mutate_dense<accrow_, selection_>(ptr, vbuffer, i, this);
262 raw.value = vbuffer;
263 }
264
265 if (report_index) {
266 if constexpr(selection_ == DimensionSelectionType::FULL) {
267 std::iota(ibuffer, ibuffer + this->full_length, 0);
268 } else if constexpr(selection_ == DimensionSelectionType::BLOCK) {
269 std::iota(ibuffer, ibuffer + this->block_length, this->block_start);
270 } else {
271 auto xptr = this->internal->index_start();
272 std::copy(xptr, xptr + this->index_length, ibuffer);
273 }
274 raw.index = ibuffer;
275 }
276
277 return raw;
278 }

280 protected:
281 bool report_index = false;
282 bool report_value = false;

289 template<bool accrow_, DimensionSelectionType selection_, bool sparse_, typename ... Args_>
290 std::unique_ptr<Extractor<selection_, sparse_, Value_, Index_> > propagate(const Options& opt, Args_... args) const {
291 std::unique_ptr<Extractor<selection_, sparse_, Value_, Index_> > output;
292
293 if constexpr(!sparse_) {
294 auto inner = new_extractor<accrow_, false>(mat.get(), opt, std::move(args)...);
295 output.reset(new DenseIsometricExtractor<accrow_, selection_>(this, std::move(inner)));
296
297 } else if constexpr(!Operation_::sparse_) {
298 bool report_value = opt.sparse_extract_value;
299 bool report_index = opt.sparse_extract_index;
300 auto inner = new_extractor<accrow_, false>(mat.get(), opt, std::move(args)...);
301 output.reset(new DensifiedSparseIsometricExtractor<accrow_, selection_>(this, std::move(inner), report_index, report_value));
302
303 } else if constexpr((accrow_ && !Operation_::needs_column_) || (!accrow_ && !Operation_::needs_row_)) { // i.e., extraction indices aren't required for the operation.
304 auto inner = new_extractor<accrow_, true>(mat.get(), opt, std::move(args)...);
305 output.reset(new SimpleSparseIsometricExtractor<accrow_, selection_>(this, std::move(inner)));
306
307 } else {
308 bool report_value = opt.sparse_extract_value;
309 bool report_index = opt.sparse_extract_index;
310
311 std::unique_ptr<Extractor<selection_, true, Value_, Index_> > inner;
312 if (report_value && !report_index) {
313 auto optcopy = opt;
314 optcopy.sparse_extract_index = true; // if we get to this clause, we need the indices; otherwise we'd have constructed a SimpleSparseIsometricExtractor.
315 inner = new_extractor<accrow_, true>(mat.get(), optcopy, std::move(args)...);
316 } else {
317 inner = new_extractor<accrow_, true>(mat.get(), opt, std::move(args)...);
318 }
319
320 output.reset(new RegularSparseIsometricExtractor<accrow_, selection_>(this, std::move(inner), report_index, report_value));
321 }
322
323 return output;
324 }
