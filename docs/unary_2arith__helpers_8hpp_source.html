33template<DelayedArithOp op_, bool right_, typename Value_, typename Scalar_>
34bool delayed_arith_actual_sparse(Scalar_ scalar) {
35 if constexpr(delayed_arith_unsupported_division_by_zero<op_, right_, Value_, Scalar_>()) {
36 // If we didn't catch this case, the else() condition would be dividing
37 // by zero in a Value_ that doesn't support it, and that would be visible
38 // at compile time - possibly resulting in compiler warnings. So we
39 // declare that this is always non-sparse, and hope that the equivalent
40 // zero() method doesn't get called.
41 return false;
42 } else {
43 // Empirically testing this, to accommodate special values (e.g., NaN, Inf) for scalars.
44 Value_ output = 0;
45 delayed_arith_run<op_, right_>(output, scalar);
46 return output == 0;
47 }
48}

50template<DelayedArithOp op_, bool right_, typename Value_, typename Scalar_>
51Value_ delayed_arith_zero(Scalar_ scalar) {
52 if constexpr(delayed_arith_unsupported_division_by_zero<op_, right_, Value_, Scalar_>()) {
53 // Avoid potential problems with division by zero that can be detected
54 // at compile time (e.g., resulting in unnecessary compiler warnings).
55 throw std::runtime_error("division by zero is not supported with IEEE-754 floats");
56 } else {
57 Value_ output = 0;
58 delayed_arith_run<op_, right_>(output, scalar);
59 return output;
60 }
61}
