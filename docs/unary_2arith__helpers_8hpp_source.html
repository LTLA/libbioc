31template<DelayedArithOp op_, bool right_, typename Value_, typename Scalar_>
32bool delayed_arith_actual_sparse(Scalar_ scalar) {
33 if constexpr(delayed_arith_unsupported_division_by_zero<op_, right_, Value_, Scalar_>()) {
34 // If we didn't catch this case, the else() condition would be dividing
35 // by zero in a Value_ that doesn't support it, and that would be visible
36 // at compile time - possibly resulting in compiler warnings. So we
37 // declare that this is always non-sparse, and hope that the equivalent
38 // zero() method doesn't get called.
39 return false;
40 } else {
41 // Empirically testing this, to accommodate special values (e.g., NaN, Inf) for scalars.
42 Value_ output = 0;
43 delayed_arith_run<op_, right_>(output, scalar);
44 return output == 0;
45 }
46}

48template<DelayedArithOp op_, bool right_, typename Value_, typename Scalar_>
49Value_ delayed_arith_zero(Scalar_ scalar) {
50 if constexpr(delayed_arith_unsupported_division_by_zero<op_, right_, Value_, Scalar_>()) {
51 // Avoid potential problems with division by zero that can be detected
52 // at compile time (e.g., resulting in unnecessary compiler warnings).
53 throw std::runtime_error("division by zero is not supported with IEEE-754 floats");
54 } else {
55 Value_ output = 0;
56 delayed_arith_run<op_, right_>(output, scalar);
57 return output;
58 }
