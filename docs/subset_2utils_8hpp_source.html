38template<bool WORKROW, typename T, typename IDX, class InputWorkspace>
39SparseRange<T, IDX> remap_sparse_duplicates(
40 const Matrix<T, IDX>* mat,
41 size_t i,
42 T* vbuffer,
43 IDX* ibuffer,
44 InputWorkspace* work,
45 const std::vector<std::pair<size_t, size_t> >& dups,
46 const std::vector<IDX>& pool)
47{
48 // Allocation status of work->vbuffer depends on the extraction mode used to construct work->internal.
49 T* vin = work->vbuffer.data();
50
51 // work->ibuffer should always be allocated, as we need this to get the expanded counts.
52 IDX* iin = work->ibuffer.data();
53 auto raw = extract_sparse<WORKROW>(mat, i, vin, iin, work->internal.get());
54
55 if (!raw.value) {
56 vbuffer = NULL;
57 }
58 if (!(work->report_index)) {
59 ibuffer = NULL;
60 }
61
62 auto vcopy = vbuffer;
63 auto icopy = ibuffer;
64 size_t counter = 0;
65
66 for (size_t i = 0; i < raw.number; ++i) {
67 const auto& pool_pos = dups[raw.index[i]];
68 counter += pool_pos.second;
69
70 if (vcopy) {
71 std::fill(vcopy, vcopy + pool_pos.second, raw.value[i]);
72 vcopy += pool_pos.second;
73 }
74
75 if (icopy) {
76 auto istart = pool.begin() + pool_pos.first;
77 std::copy(istart, istart + pool_pos.second, icopy);
78 icopy += pool_pos.second;
79 }
80 }
81
82 return SparseRange<T, IDX>(counter, vbuffer, ibuffer);
83}
