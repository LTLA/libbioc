23void delayed_boolean_cast(const InputValue_* input, Index_ length, OutputValue_* output) {
24 SUBPAR_VECTORIZABLE
25 for (Index_ i = 0; i < length; ++i) {
26 if constexpr(std::is_same<InputValue_, OutputValue_>::value) {
27 auto& val = output[i];
28 val = static_cast<bool>(val);
29 } else {
30 output[i] = static_cast<bool>(input[i]);
31 }
32 }
33}

48template<BooleanOperation op_, typename InputValue_, typename Index_, typename OutputValue_>
49void delayed_boolean_run_simple(const InputValue_* input, Index_ length, bool scalar, OutputValue_* output) {
50 if constexpr(op_ == BooleanOperation::AND) {
51 if (scalar) {
52 delayed_boolean_cast(input, length, output);
53 } else {
54 std::fill_n(output, length, 0);
55 }
56 } else if constexpr(op_ == BooleanOperation::OR) {
57 if (scalar) {
58 std::fill_n(output, length, 1);
59 } else {
60 delayed_boolean_cast(input, length, output);
61 }
62 } else if constexpr(op_ == BooleanOperation::XOR) {
63 if (scalar) {
64 delayed_boolean_not(input, length, output);
65 } else {
66 delayed_boolean_cast(input, length, output);
67 }
68 } else { // EQUAL
69 if (scalar) {
70 delayed_boolean_cast(input, length, output);
71 } else {
72 delayed_boolean_not(input, length, output);
73 }
74 }
75}
