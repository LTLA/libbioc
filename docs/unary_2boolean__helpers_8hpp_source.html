19template<typename InputValue_, typename Index_, typename OutputValue_>
20void delayed_boolean_cast(const InputValue_* input, Index_ length, OutputValue_* output) {
21#ifdef _OPENMP
22 #pragma omp simd
23#endif
24 for (Index_ i = 0; i < length; ++i) {
25 if constexpr(std::is_same<InputValue_, OutputValue_>::value) {
26 auto& val = output[i];
27 val = static_cast<bool>(val);
28 } else {
29 output[i] = static_cast<bool>(input[i]);

31 }

49template<BooleanOperation op_, typename InputValue_, typename Index_, typename OutputValue_>
50void delayed_boolean_run_simple(const InputValue_* input, Index_ length, bool scalar, OutputValue_* output) {
51 if constexpr(op_ == BooleanOperation::AND) {
52 if (scalar) {
53 delayed_boolean_cast(input, length, output);
54 } else {
55 std::fill_n(output, length, 0);
56 }
57 } else if constexpr(op_ == BooleanOperation::OR) {
58 if (scalar) {
59 std::fill_n(output, length, 1);
60 } else {
61 delayed_boolean_cast(input, length, output);
62 }
63 } else if constexpr(op_ == BooleanOperation::XOR) {
64 if (scalar) {
65 delayed_boolean_not(input, length, output);
66 } else {
67 delayed_boolean_cast(input, length, output);
68 }
69 } else { // EQUAL
70 if (scalar) {
71 delayed_boolean_cast(input, length, output);
72 } else {
73 delayed_boolean_not(input, length, output);
74 }
75 }
76}
