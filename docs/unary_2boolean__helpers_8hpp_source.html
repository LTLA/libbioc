26template<DelayedBooleanOp op_>
27bool delayed_boolean_actual_sparse(bool scalar) {
28 if constexpr(op_ ==DelayedBooleanOp::OR || op_ == DelayedBooleanOp::XOR) {
29 // for both or and xor, if the scalar is true, then
