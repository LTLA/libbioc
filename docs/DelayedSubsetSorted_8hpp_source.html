123 struct SparseBase {
124 /*
125 * In general, we can't extract indices/values directly into the user-supplied ibuffer/vbuffer,
126 * because we need some place to hold the to-be-processed values while we expand the duplicates.
127 * Thus, we need to allocate the workspace's ibuffer/vbuffer to the length of the unique indices.
128 *
129 * Obviously, if the values are not required, we can skip the allocation of vbuffer.
130 * However, ibuffer is still required, even if the indices are not required, to figure out the expansion of each value.
131 * In fact, even if both are not required, ibuffer is still required because we still need the number of times each value is expanded to get the count.
132 */
133
134 SparseBase(size_t bufsize, const WorkspaceOptions& opt) :
135 report_index(sparse_extract_index(opt.sparse_extract_mode)),
136 vbuffer(sparse_extract_value(opt.sparse_extract_mode) ? bufsize : 0),
137 ibuffer(bufsize)
138 {}
139
140 bool report_index;
141 std::vector<T> vbuffer;
142 std::vector<IDX> ibuffer;
143 };

290 template<bool SPARSE, class Function>
291 std::vector<IDX> transplant_indices(subset_utils::ConditionalSupplement<IDX, SPARSE>& temp, size_t length, Function to_index) const {
292 if constexpr(SPARSE) {
293 size_t mapping_dim = MARGIN == 0 ? mat->nrow() : mat->ncol();
294 temp.mapping_duplicates.resize(mapping_dim);
295 temp.mapping_duplicates_pool.reserve(length);
296 } else {
297 temp.reverse_mapping.reserve(length);
298 }
299
300 std::vector<IDX> local;
301 local.reserve(length);
302
303 for (size_t x = 0; x < length; ++x) {
304 auto i = to_index(x);
305 auto s = indices[i];
306 bool diff = local.empty() || s != local.back();
307 if (diff) {
308 local.push_back(s);
309 }
310
311 if constexpr(SPARSE) {
312 auto& range = temp.mapping_duplicates[s];
313 if (diff) {
314 range.first = temp.mapping_duplicates_pool.size();
315 }
316 ++range.second;
317 temp.mapping_duplicates_pool.push_back(i);
318 } else {
319 temp.reverse_mapping.push_back(local.size() - 1);
320 }
321 }
322
323 return local;
324 }
