579 static void copy_over(SparseRange<T, IDX>& output, T* vbuffer, IDX* ibuffer, SparseCopyMode copy) {
580 if ((copy == SPARSE_COPY_BOTH || copy == SPARSE_COPY_INDEX) && output.index != ibuffer) {

586 copy_over(output.value, vbuffer, output.number);
587 output.value = vbuffer;
