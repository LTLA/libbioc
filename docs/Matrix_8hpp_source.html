585 static void copy_over(SparseRange<T, IDX>& output, T* vbuffer, IDX* ibuffer, SparseCopyMode copy) {
586 if ((copy == SPARSE_COPY_BOTH || copy == SPARSE_COPY_INDEX) && output.index != ibuffer) {
587 copy_over(output.index, ibuffer, output.number);
588 output.index = ibuffer;
589 }
590
591 if ((copy == SPARSE_COPY_BOTH || copy == SPARSE_COPY_VALUE) && output.value != vbuffer) {
592 copy_over(output.value, vbuffer, output.number);
593 output.value = vbuffer;
594 }
595 }
