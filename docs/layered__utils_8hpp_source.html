19template<typename T>
20int categorize_row(T v) {
21 constexpr uint8_t max8 = std::numeric_limits<uint8_t>::max();
22 constexpr uint16_t max16 = std::numeric_limits<uint16_t>::max();
23 constexpr T maxv = std::numeric_limits<T>::max();
24
25 if constexpr(maxv <= max8) {
26 return 0;
27 } else {
28 if (v <= max8) {
29 return 0;
30 }
31 }
32
33 if constexpr(maxv <= max16) {
34 return 1;
35 } else {
36 if (v <= max16) {
37 return 1;
38 }
39 }
40
41 return 2;
42}

54template<typename O>
55RowRemapping<O> compute_new_indices(const std::vector<int>& category) {
56 RowRemapping<O> output(category.size());
57 for (size_t i = 0; i < category.size(); ++i) {
58 auto& sofar = output.per_category[category[i]];
59 output.new_indices[i] = sofar;
60 ++sofar;
61 }
62
63 // Computing the permutation, while we're here.
64 O offset[3];
65 offset[0] = 0;
66 offset[1] = output.per_category[0];
67 offset[2] = offset[1] + output.per_category[1];
68
69 for (size_t i = 0; i < category.size(); ++i){
70 output.permutation[i] = offset[category[i]] + output.new_indices[i];
71 }
72
73 return output;
74}

76template<typename O>
77RowRemapping<O> create_empty(size_t n) {
78 RowRemapping<O> output(n);
79 std::iota(output.new_indices.begin(), output.new_indices.end(), 0);
80 std::iota(output.permutation.begin(), output.permutation.end(), 0);
81 return output;
