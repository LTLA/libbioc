27template<DelayedBooleanOp op_, typename Value_>
28void delayed_boolean_run(Value_& val, bool scalar) {
29 if constexpr(op_ == DelayedBooleanOp::AND) {
30 val = val && scalar;
31 } else if constexpr(op_ == DelayedBooleanOp::OR) {
32 val = val || scalar;
33 } else if constexpr(op_ == DelayedBooleanOp::XOR) {
34 val = static_cast<bool>(val) != scalar;
35 } else { // EQUAL.
36 val = static_cast<bool>(val) == scalar;
37 }
38}

47template<DelayedBooleanOp op_>
48bool delayed_boolean_actual_sparse(bool scalar) {
49 if constexpr(op_ ==DelayedBooleanOp::OR || op_ == DelayedBooleanOp::XOR) {
50 // for both or and xor, if the scalar is true, then
51 // the result will not be sparse, as zeros + scalar will be true.
52 return !scalar;
53 } else { // i.e., EQUAL.
54 return scalar;
55 }
