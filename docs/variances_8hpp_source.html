61template<typename Output_ = double, typename Value_>
62std::pair<Output_, Output_> compute_direct(const Value_* ptr, size_t n) {
63 if (n < 1) {
64 return both_NaN<Output_>();
65 }
66
67 Output_ mean = std::accumulate(ptr, ptr + n, static_cast<Output_>(0))/n;
68 Output_ var = 0;
69 for (size_t j = 0; j < n; ++j, ++ptr) {
70 var += (*ptr - mean) * (*ptr - mean);
71 }
72
73 return std::make_pair(mean, finish_variance_direct(var, n));
74}

90template<typename Output_ = double, typename Value_, typename Index_>
91std::pair<Output_, Output_> compute_direct(const SparseRange<Value_, Index_>& range, size_t n) {
92 if (n < 1) {
93 return both_NaN<Output_>();
94 }
95
96 Output_ mean = std::accumulate(range.value, range.value + range.number, static_cast<Output_>(0))/n;
97 Output_ var = 0;
98 for (Index_ j = 0; j < range.number; ++j) {
99 var += (range.value[j] - mean) * (range.value[j] - mean);
100 }
101 var += mean * mean * (n - range.number);
102

155template<typename Value_, typename Index_, typename Output_, typename Nonzero_>
156void compute_running(const SparseRange<Value_, Index_>& range, Output_* means, Output_* vars, Nonzero_* nonzeros, int& count, bool skip_zeros = true, Index_ subtract = 0) {
157 ++count;
158 for (Index_ j = 0; j < range.number; ++j) {
159 if (!skip_zeros || range.value[j]) {
160 auto ri = range.index[j] - subtract;
161 auto& curM = means[ri];
162 auto& curS = vars[ri];
163 auto& curNZ = nonzeros[ri];
164 ++curNZ;
165
166 const auto& curval = range.value[j];
167 const double delta = curval - curM;
168 curM += delta / curNZ;
169 curS += delta * (curval - curM);
170 }
