60template<typename O = double, typename T>
61std::pair<O, O> compute_direct(const T* ptr, size_t n) {
62 if (n < 1) {
63 return both_NaN<O>();
64 }
65
66 O mean = std::accumulate(ptr, ptr + n, static_cast<O>(0))/n;
67 O var = 0;
68 for (size_t j = 0; j < n; ++j, ++ptr) {
69 var += (*ptr - mean) * (*ptr - mean);
70 }
71
72 return std::make_pair(mean, finish_variance_direct(var, n));
73}

89template<typename O = double, typename T, typename IDX>
90std::pair<O, O> compute_direct(const SparseRange<T, IDX>& range, size_t n) {
91 if (n < 1) {
92 return both_NaN<O>();
93 }
94
95 O mean = std::accumulate(range.value, range.value + range.number, static_cast<O>(0))/n;
96 O var = 0;
97 for (size_t j = 0; j < range.number; ++j) {
98 var += (range.value[j] - mean) * (range.value[j] - mean);
99 }
100 var += mean * mean * (n - range.number);
101
102 return std::make_pair(mean, finish_variance_direct(var, n));
103}

152template<typename T, typename IDX, typename O, typename Nz>
153void compute_running(const SparseRange<T, IDX>& range, O* means, O* vars, Nz* nonzeros, int& count) {
154 ++count;
155 for (size_t j = 0; j < range.number; ++j) {
156 auto ri = range.index[j];
157 auto& curM = means[ri];
158 auto& curS = vars[ri];
159 auto& curNZ = nonzeros[ri];
160 ++curNZ;
161
162 const auto& curval = range.value[j];
163 const double delta = curval - curM;
164 curM += delta / curNZ;
165 curS += delta * (curval - curM);
166 }
167}

182template<typename O>
183void finish_running(size_t n, O* means, O* vars, int count) {
184 if (count > 1) {
185 for (size_t i = 0; i < n; ++i) {
186 vars[i] /= count - 1;

211template<typename O, typename Nz>
212void finish_running(size_t n, O* means, O* vars, Nz* nonzeros, int count) {
213 if (count) {
214 for (size_t i = 0; i < n; ++i) {
215 const double curNZ = nonzeros[i];
216 const double ratio = curNZ / count;
217 auto& curM = means[i];
218 vars[i] += curM * curM * ratio * (count - curNZ);
219 curM *= ratio;
220 }
221 }
222 finish_running(n, means, vars, count);
