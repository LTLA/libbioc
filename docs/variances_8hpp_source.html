63template<typename Output_ = double, typename Value_>
64std::pair<Output_, Output_> compute_direct(const Value_* ptr, size_t n) {
65 if (n < 1) {
66 return both_NaN<Output_>();
67 }
68
69 Output_ mean = std::accumulate(ptr, ptr + n, static_cast<Output_>(0))/n;
70 Output_ var = 0;
71 for (size_t j = 0; j < n; ++j, ++ptr) {
72 var += (*ptr - mean) * (*ptr - mean);
73 }
74
75 return std::make_pair(mean, finish_variance_direct(var, n));
76}

92template<typename Output_ = double, typename Value_, typename Index_>
93std::pair<Output_, Output_> compute_direct(const SparseRange<Value_, Index_>& range, size_t n) {
94 if (n < 1) {
95 return both_NaN<Output_>();
96 }
97
98 Output_ mean = std::accumulate(range.value, range.value + range.number, static_cast<Output_>(0))/n;
99 Output_ var = 0;
100 for (Index_ j = 0; j < range.number; ++j) {
101 var += (range.value[j] - mean) * (range.value[j] - mean);
102 }
103 var += mean * mean * (n - range.number);

157template<typename Value_, typename Index_, typename Output_, typename Nonzero_>
158void compute_running(const SparseRange<Value_, Index_>& range, Output_* means, Output_* vars, Nonzero_* nonzeros, int& count, bool skip_zeros = true, Index_ subtract = 0) {
159 ++count;
160 for (Index_ j = 0; j < range.number; ++j) {
161 if (!skip_zeros || range.value[j]) {
162 auto ri = range.index[j] - subtract;
163 auto& curM = means[ri];
164 auto& curS = vars[ri];
165 auto& curNZ = nonzeros[ri];
166 ++curNZ;
167

170 curM += delta / curNZ;
