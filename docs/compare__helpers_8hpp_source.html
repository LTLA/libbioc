29template<DelayedCompareOp op_, typename Scalar_, typename Value_>
30void delayed_compare_run(Value_& val, Scalar_ scalar) {
31 if constexpr(op_ == DelayedCompareOp::EQUAL) {
32 val = val == scalar;
33 } else if constexpr(op_ == DelayedCompareOp::GREATER_THAN) {
34 val = val > scalar;
35 } else if constexpr(op_ == DelayedCompareOp::LESS_THAN) {
36 val = val < scalar;
37 } else if constexpr(op_ == DelayedCompareOp::GREATER_THAN_OR_EQUAL) {
38 val = val >= scalar;
39 } else if constexpr(op_ == DelayedCompareOp::LESS_THAN_OR_EQUAL) {
40 val = val <= scalar;
41 } else { // NOT EQUAL.
42 val = val != scalar;
43 }
44}

53template<DelayedCompareOp op_, typename Scalar_>
54bool delayed_compare_actual_sparse(Scalar_ scalar) {
55 if constexpr(op_ == DelayedCompareOp::EQUAL) {
