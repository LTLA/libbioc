129 DelayedSubsetSortedUnique(std::shared_ptr<const Matrix<Value_, Index_> > p, IndexStorage_ idx, bool row, bool check = true) :
130 mat(std::move(p)), indices(std::move(idx)), by_row(row)
131 {
132 if (check) {
133 for (Index_ i = 1, end = indices.size(); i < end; ++i) {
134 if (indices[i] <= indices[i-1]) {
135 throw std::runtime_error("indices should be unique and sorted");
136 }
137 }
138 }
139
140 Index_ mapping_dim = by_row ? mat->nrow() : mat->ncol();
141 mapping_single.resize(mapping_dim);
142 for (Index_ i = 0, end = indices.size(); i < end; ++i) {
143 mapping_single[indices[i]] = i;
144 }
145 }
