136 return extract_sparse<WORKROW>(mat.get(), indices[i], out_values, out_indices, work);

141 // Only updating the indices if we actually extracted some;

144 auto icopy = out_indices;
145 for (size_t i = 0; i < raw.number; ++i, ++icopy) {
146 *icopy = mapping_single[raw.index[i]];

150 }
