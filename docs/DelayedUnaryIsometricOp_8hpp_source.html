243 if (range.value != vbuffer) {
244 // Don't use fetch_copy as we don't want to copy the indices.
245 std::copy(range.value, range.value + range.number, vbuffer);

248

250 if (range.number < full_length) { // avoid calling zero() if possible, as this might throw zero-related errors in non-IEEE platforms.

256 } else if (this->parent->operation.actual_sparse()) {
257 return static_cast<Value_>(0);
258 } else {

260 }
261 }());
262 }
263
264 if constexpr(selection_ == DimensionSelectionType::FULL) {
265 for (Index_ i = 0; i < range.number; ++i) {
266 buffer[range.index[i]] = vbuffer[i];

268 } else if constexpr(selection_ == DimensionSelectionType::BLOCK) {
269 auto shift = this->internal->block_start;
270 for (Index_ i = 0; i < range.number; ++i) {
271 buffer[range.index[i] - shift] = vbuffer[i];

273 } else {

548 template<bool accrow_, DimensionSelectionType selection_, bool sparse_, typename ... Args_>
549 std::unique_ptr<Extractor<selection_, sparse_, Value_, Index_> > propagate(const Options& opt, Args_&& ... args) const {
550 std::unique_ptr<Extractor<selection_, sparse_, Value_, Index_> > output;
551
552 if constexpr(!sparse_) {
553 if (!(mat->sparse())) {
554 output.reset(new DenseIsometricExtractor_Basic<accrow_, selection_>(this, opt, std::forward<Args_>(args)...));
555
556 } else {
557 if (preserves_sparsity()) {
558 output.reset(new DenseIsometricExtractor_FromSparse<accrow_, selection_>(this, opt, std::forward<Args_>(args)...));
559
560 } else {
561 if constexpr(accrow_ && !Operation_::needs_column) { // constant sparsity breaking
562 output.reset(new DenseIsometricExtractor_FromSparse<accrow_, selection_>(this, opt, std::forward<Args_>(args)...));
563 } else if constexpr(!accrow_ && !Operation_::needs_row) { // constant sparsity breaking
564 output.reset(new DenseIsometricExtractor_FromSparse<accrow_, selection_>(this, opt, std::forward<Args_>(args)...));
565 } else { // variable sparsity breaking
566 output.reset(new DenseIsometricExtractor_Basic<accrow_, selection_>(this, opt, std::forward<Args_>(args)...));
567 }
568 }
569 }
570
571 } else {
572 if (!(mat->sparse())) {
573 output.reset(new SparseIsometricExtractor_FromDense<accrow_, selection_>(this, opt, std::forward<Args_>(args)...));
574
575 } else {
576 if (preserves_sparsity()) {
577 if constexpr(accrow_ && !Operation_::needs_column) { // no need for column indices
578 output.reset(new SparseIsometricExtractor_Simple<accrow_, selection_>(this, opt, std::forward<Args_>(args)...));
579 } else if constexpr(!accrow_ && !Operation_::needs_row) { // no need for row indices
580 output.reset(new SparseIsometricExtractor_Simple<accrow_, selection_>(this, opt, std::forward<Args_>(args)...));
581 } else { // guess we need indices
582 output.reset(new SparseIsometricExtractor_NeedsIndices<accrow_, selection_>(this, opt, std::forward<Args_>(args)...));
583 }
584
585 } else {
586 if constexpr(accrow_ && !Operation_::needs_column) { // constant sparsity breaking
587 output.reset(new SparseIsometricExtractor_ForcedDense<accrow_, selection_>(this, opt, std::forward<Args_>(args)...));
588 } else if constexpr(!accrow_ && !Operation_::needs_row) { // constant sparsity breaking
589 output.reset(new SparseIsometricExtractor_ForcedDense<accrow_, selection_>(this, opt, std::forward<Args_>(args)...));
590 } else { // variable sparsity breaking
591 output.reset(new SparseIsometricExtractor_FromDense<accrow_, selection_>(this, opt, std::forward<Args_>(args)...));
592 }
593 }
594 }
595 }
596
597 return output;
