19template<CompareOperation op_, typename Value_, typename Scalar_>
20bool delayed_substitute_is_sparse(Value_ compared, Scalar_ substitute) {
21 return !delayed_compare<op_, Value_>(0, compared) || substitute == 0;
22}

24template<CompareOperation op_, typename Value_, typename Scalar_>
25void delayed_substitute_run(Value_& val, Value_ compared, Scalar_ substitute) {
26 if (delayed_compare<op_, Value_>(val, compared)) {
27 val = substitute;
28 }
29}

31template<CompareOperation op_, typename Value_, typename Index_, typename Scalar_>
32void delayed_substitute_run_simple(Value_* buffer, Index_ length, Value_ compared, Scalar_ substitute) {
33 for (Index_ i = 0; i < length; ++i) {
34 delayed_substitute_run<op_, Value_>(buffer[i], compared, substitute);

121template<CompareOperation op_, typename Value_ = double, typename Vector_ = std::vector<Value_> >
122class DelayedUnaryIsometricSubstituteVector {
123public:
134 DelayedUnaryIsometricSubstituteVector(Vector_ compared, Vector_ substitute, bool by_row) :
135 my_compared(std::move(compared)), my_substitute(std::move(substitute)), my_by_row(by_row)
136 {
137 for (size_t i = 0, end = my_compared.size(); i < end; ++i) {
138 if (!delayed_substitute_is_sparse<op_, Value_>(my_compared[i], my_substitute[i])) {
139 my_sparse = false;
140 break;
141 }
142 }
143 }
144
145private:
146 Vector_ my_compared, my_substitute;
147 bool my_by_row;
148 bool my_sparse = true;
149
150public:
154 static constexpr bool is_basic = false;
155
156 bool zero_depends_on_row() const {
157 return my_by_row;
158 }
159
160 bool zero_depends_on_column() const {
161 return !my_by_row;
162 }
163
164 bool non_zero_depends_on_row() const {
165 return my_by_row;
166 }
167
168 bool non_zero_depends_on_column() const {
169 return !my_by_row;
170 }
171
172 bool is_sparse() const {
173 return my_sparse;
174 }
179public:
183 template<typename Index_>
184 void dense(bool row, Index_ idx, Index_ start, Index_ length, Value_* buffer) const {
185 if (row == my_by_row) {
186 delayed_substitute_run_simple<op_, Value_>(buffer, length, my_compared[idx], my_substitute[idx]);
187 } else {
188 for (Index_ i = 0; i < length; ++i) {
189 Index_ is = i + start;
190 delayed_substitute_run<op_, Value_>(buffer[i], my_compared[is], my_substitute[is]);
191 }
192 }
193 }

196 void dense(bool row, Index_ idx, const std::vector<Index_>& indices, Value_* buffer) const {

199 } else {

203 }

208 void sparse(bool row, Index_ idx, Index_ number, Value_* buffer, const Index_* indices) const {

211 } else {
212 for (Index_ i = 0; i < number; ++i) {
213 auto ii = indices[i];

215 }
216 }
217 }

219 template<typename Index_>
220 Value_ fill(bool row, Index_ idx) const {
221 if (row == my_by_row) {
222 auto sub = my_substitute[idx];
223 if (!delayed_compare<op_, Value_>(0, my_compared[idx])) {
224 return 0;
225 } else {
226 return sub;
227 }
228 } else {
229 // We should only get to this point if it's sparse, otherwise no
230 // single fill value would work across the length of my_compared.
231 return 0;
232 }
