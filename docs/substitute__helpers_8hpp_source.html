180 return my_sparse;
181 }

190 template<typename Index_>
191 void dense(bool row, Index_ idx, Index_ start, Index_ length, const Value_*, Value_* output) const {
192 if (row == my_by_row) {
193 delayed_substitute_run_simple<op_, Value_>(output, length, my_compared[idx], my_substitute[idx]);
194 } else {

199 }

230 template<typename, typename, typename Index_>
231 Value_ fill(bool row, Index_ idx) const {
232 if (row == my_by_row) {
233 auto sub = my_substitute[idx];
234 if (!delayed_compare<op_, Value_>(0, my_compared[idx])) {
235 return 0;
236 } else {
237 return sub;
238 }
239 } else {
240 // We should only get to this point if it's sparse, otherwise no
241 // single fill value would work across the length of my_compared.
242 return 0;
243 }
