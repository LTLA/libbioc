120template<CompareOperation op_, typename Value_ = double, typename Vector_ = std::vector<Value_> >
121class DelayedUnaryIsometricSubstituteVector {
122public:
133 DelayedUnaryIsometricSubstituteVector(Vector_ compared, Vector_ substitute, bool by_row) :
134 my_compared(std::move(compared)), my_substitute(std::move(substitute)), my_by_row(by_row)
135 {
136 for (size_t i = 0, end = my_compared.size(); i < end; ++i) {
137 if (!delayed_substitute_is_sparse<op_, Value_>(my_compared[i], my_substitute[i])) {
138 my_sparse = false;
139 break;
140 }
141 }
142 }
143
144private:
145 Vector_ my_compared, my_substitute;
146 bool my_by_row;
147 bool my_sparse = true;
148
149public:
153 static constexpr bool is_basic = false;
154
155 bool zero_depends_on_row() const {
156 return my_by_row;
157 }
158
159 bool zero_depends_on_column() const {
160 return !my_by_row;
161 }
162
163 bool non_zero_depends_on_row() const {
164 return my_by_row;
165 }
166
167 bool non_zero_depends_on_column() const {
168 return !my_by_row;
169 }
170
171 bool is_sparse() const {
172 return my_sparse;
173 }
178public:
182 template<typename Index_>
183 void dense(bool row, Index_ idx, Index_ start, Index_ length, Value_* buffer) const {
184 if (row == my_by_row) {
185 delayed_substitute_run_simple<op_, Value_>(buffer, length, my_compared[idx], my_substitute[idx]);
186 } else {
187 for (Index_ i = 0; i < length; ++i) {
188 Index_ is = i + start;
189 delayed_substitute_run<op_, Value_>(buffer[i], my_compared[is], my_substitute[is]);
190 }
191 }
192 }
193
194 template<typename Index_>
195 void dense(bool row, Index_ idx, const std::vector<Index_>& indices, Value_* buffer) const {
196 if (row == my_by_row) {
197 delayed_substitute_run_simple<op_, Value_>(buffer, static_cast<Index_>(indices.size()), my_compared[idx], my_substitute[idx]);
198 } else {
199 for (Index_ i = 0, length = indices.size(); i < length; ++i) {
200 auto ii = indices[i];
201 delayed_substitute_run<op_, Value_>(buffer[i], my_compared[ii], my_substitute[ii]);
202 }
203 }
204 }

206 template<typename Index_>
207 void sparse(bool row, Index_ idx, Index_ number, Value_* buffer, const Index_* indices) const {
208 if (row == my_by_row) {
209 delayed_substitute_run_simple<op_, Value_>(buffer, number, my_compared[idx], my_substitute[idx]);
210 } else {
211 for (Index_ i = 0; i < number; ++i) {
212 auto ii = indices[i];
213 delayed_substitute_run<op_, Value_>(buffer[i], my_compared[ii], my_substitute[ii]);
214 }
215 }
216 }
217
218 template<typename Index_>
219 Value_ fill(bool row, Index_ idx) const {
220 if (row == my_by_row) {
221 auto sub = my_substitute[idx];
222 if (!delayed_compare<op_, Value_>(0, my_compared[idx])) {
223 return 0;
224 } else {
225 return sub;
226 }
227 } else {
228 // We should only get to this point if it's sparse, otherwise no
229 // single fill value would work across the length of my_compared.
230 return 0;
231 }
232 }
