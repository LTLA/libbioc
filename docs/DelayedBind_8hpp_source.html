431 if constexpr((MARGIN == 0) == WORKROW) {
432 std::vector<std::shared_ptr<Parent> > workspaces;
433 workspaces.reserve(mats.size());
434 for (const auto& x : mats) {
435 workspaces.push_back(new_workspace<WORKROW, Parent::sparse>(x.get(), start, length, opt));
436 }

438
439 } else {
440 std::vector<std::shared_ptr<Parent> > workspaces;
441 std::vector<size_t> kept;
442
443 if (length) {
444 workspaces.reserve(mats.size());
445 kept.reserve(mats.size());
446
447 size_t index = std::upper_bound(cumulative.begin(), cumulative.end(), start) - cumulative.begin() - 1; // finding the first matrix.
448 size_t actual_start = start - cumulative[index];
449 size_t end = start + length;
450
451 for (; index < mats.size(); ++index) {
452 size_t actual_end = end;
453 bool not_final = (end > cumulative[index + 1]);
454 if (not_final) {
455 actual_end = cumulative[index + 1];
456 }
457 actual_end -= cumulative[index];
458
459 size_t len = actual_end - actual_start;
460 const auto& x = mats[index];
461 workspaces.push_back(new_workspace<WORKROW, Parent::sparse>(x.get(), actual_start, len, opt));
462 kept.push_back(index);
463
464 if (!not_final) {
465 break;
466 }
467 actual_start = 0;
468 }
469 }

638 std::shared_ptr<Parent> output(ptr);
639 const auto& subset = ptr->indices_;
640 size_t length = subset.size();
641
642 if (!subset.empty()) {
643 auto& workspaces = ptr->workspaces;
644 auto& kept = ptr->kept;
645
646 workspaces.reserve(mats.size());
647 kept.reserve(mats.size());
648
649 size_t index = std::upper_bound(cumulative.begin(), cumulative.end(), subset[0]) - cumulative.begin() - 1; // finding the first matrix.
650 size_t counter = 0;
651
652 for (; index < mats.size(); ++index) {
653 IDX lower = cumulative[index];
654 IDX upper = cumulative[index + 1];
655
656 std::vector<IDX> curslice;
657 while (counter < length && subset[counter] < upper) {
658 curslice.push_back(subset[counter] - lower);
659 ++counter;
660 }
661
662 const auto& x = mats[index];
663 workspaces.push_back(new_workspace<WORKROW, Parent::sparse>(x.get(), std::move(curslice), opt));
664 kept.push_back(index);
665
666 if (counter == length) {
667 break;
668 }
669 }
670 }
671
672 return output;
673 }
674 }
