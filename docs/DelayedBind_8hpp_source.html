559 std::shared_ptr<IndexWorkspace<IDX, WORKROW> > output(ptr);
560 ptr->workspaces.reserve(mats.size());

565 } else {
566 ptr->workspaces.push_back(x->new_column_workspace(i));

568 }
569
570 if (mats.empty()) { // make sure we can provide the indices if there are no matrices.
571 ptr->indices_ = std::move(i);

574

577 std::shared_ptr<IndexWorkspace<IDX, WORKROW> > output(ptr);
578 const auto& subset = ptr->indices_;

580

582 auto& workspaces = ptr->workspaces;
583 auto& kept = ptr->kept;
584
585 workspaces.reserve(mats.size());
586 kept.reserve(mats.size());
587
588 size_t index = std::upper_bound(cumulative.begin(), cumulative.end(), subset[0]) - cumulative.begin() - 1; // finding the first matrix.

590
591 for (; index < mats.size(); ++index) {
592 IDX lower = cumulative[index];
593 IDX upper = cumulative[index + 1];
594
595 std::vector<IDX> curslice;
596 while (counter < length && subset[counter] < upper) {
597 curslice.push_back(subset[counter] - lower);

604 } else {

606 }
607 kept.push_back(index);
608
