288 size_t fill(size_t id, Index_* buffer, size_t number) {
289 auto& current = counters[id];
290 size_t end = current + number;
291 size_t available = stream.size();
292
293 if (available >= end) {
294 std::copy(stream.begin() + current, stream.begin() + end, buffer);
295 current = end;
296 return number;
297 }
298
299 size_t handled = 0;
300 if (current < available) {
301 std::copy(stream.begin() + current, stream.end(), buffer);
302 handled = available - current;
303 buffer += handled;
304 number -= handled;
305 }
306
307 size_t filled = source->predict(buffer, number);
308 current = available + filled;
309
310 // Try to slim down if the accumulated stream has gotten too big.
311 if (stream.size() >= 10000) {
312 size_t minimum = *std::min_element(counters.begin(), counters.end());
313 if (minimum) {
314 stream.erase(stream.begin(), stream.begin() + minimum);
315 for (auto& x : counters) {
316 x -= minimum;
317 }
318 }
319 }
320
321 stream.insert(stream.end(), buffer, buffer + filled);
322 return filled + handled;
323 }

537 ParentOracle(std::unique_ptr<Oracle<Index_> > o, std::vector<unsigned char> u, const std::vector<Index_>* c) :
538 source(std::move(o)), streams(u.size()), used(std::move(u)), cumulative(c) {}
539
540 size_t fill(size_t id, Index_* buffer, size_t number) {
541 auto& stream = streams[id];
542
543 // Trying to top it up, but we won't try too hard, as we might
544 // end up doing lots of predictions to fill up the stream for a
545 // bound matrix with very few elements.
546 if (stream.size() < number) {
547 const auto& cum = *cumulative;
548 bool unfound = true;
549 do {
550 size_t filled = source->predict(buffer, number);
551 if (!filled) {
552 break;
553 }
554
555 for (size_t f = 0; f < filled; ++f) {
556 PerpendicularExtractor::choose_segment(buffer[f], chosen_segment, cum);
557 if (used[chosen_segment]) {
558 streams[chosen_segment].push_back(buffer[f] - cum[chosen_segment]);
559 if (chosen_segment == id) {
560 unfound = true;
561 }
562 }
563 }
564 } while (unfound);
565
566 if (stream.size() < number) {
567 number = stream.size();
568 }
569 }
570
571 if (number) {
572 std::copy(stream.begin(), stream.begin() + number, buffer);
573 stream.erase(stream.begin(), stream.begin() + number);
574 }
575 return number;
576 }

579 std::vector<std::deque<Index_> > streams;
