29template<typename Index_, class IndexStorage_, class ToIndex_>
30DenseParallelResults<Index_> format_dense_parallel_base(const IndexStorage_& indices, Index_ len, ToIndex_ to_index) {
31 std::vector<std::pair<Index_, Index_> > collected;
32 collected.reserve(len);
33 for (Index_ i = 0; i < len; ++i) {
34 collected.emplace_back(indices[to_index(i)], i);
35 }
36 std::sort(collected.begin(), collected.end());
37
38 DenseParallelResults<Index_> output;
39 if (collected.size()) {
40 output.collapsed.reserve(len);
41 output.reindex.resize(len);
42
43 Index_ last = collected.front().first;
44 output.collapsed.push_back(last);
45 output.reindex[collected.front().second] = 0;
46
47 Index_ counter = 0;
48 for (Index_ i = 1; i < len; ++i) {
49 const auto& pp = collected[i];
50 if (pp.first != last) {
51 last = pp.first;
52 output.collapsed.push_back(last);
53 ++counter;
54 }
55 output.reindex[pp.second] = counter;
56 }
57 }
58
59 return output;
60}

110template<typename Index_>
111struct SparseParallelReindex {
112 // This is a bit complicated to explain.
113 // Let 'x = pool_ptrs[i - offset]'.
114 // Let 'y = pool_ptrs[i - offset + 1]'.
115 // Let 'z' denote any integer in '[x, y)'.
116 // In which case, 'indices[pool_indices[z]]' is equal to 'i'.
117 // The general idea is that 'pool_indices[z]' can be used to fill the 'SparseRange::index' on output.
118 std::vector<Index_> pool_ptrs;
119 std::vector<Index_> pool_indices;
120 Index_ offset;
121};

129template<typename Index_, class IndexStorage_, class ToIndex_>
130SparseParallelResults<Index_> format_sparse_parallel_base(const IndexStorage_& indices, Index_ len, ToIndex_ to_index) {
131 std::vector<std::pair<Index_, Index_> > collected;
132 collected.reserve(len);
133 for (Index_ i = 0; i < len; ++i) {
134 auto curdex = to_index(i);
135 collected.emplace_back(indices[curdex], curdex);
136 }
137 std::sort(collected.begin(), collected.end());
138
139 SparseParallelResults<Index_> output;
140
141 if (collected.size()) {
142 output.collapsed.reserve(len);
143 output.reindex.pool_indices.reserve(len);
144 Index_ first = collected.front().first;
145
146 // 'pool_ptrs' is a vector that enables look-up according to the
147 // indices of the underlying array. to avoid the need to allocate a
148 // vector of length equal to the underlying array's dimension, we only
149 // consider the extremes of 'indices'; we allocate 'pool_ptrs' to have
150 // length equal to the range of 'indices' (plus 1, as we're storing
151 // cumulative pointers). 'offset' defines the lower bound that must be
152 // subtracted from the array indices to get an index into 'pool_ptrs'.
153 output.reindex.offset = first;
154 auto allocation = collected.back().first - output.reindex.offset + 1;
155 output.reindex.pool_ptrs.resize(allocation + 1);
156
157 Index_ counter = 0;
158 output.reindex.pool_ptrs[counter] = 0;
159 ++counter;
160 output.reindex.pool_indices.push_back(collected.front().second);
161 output.reindex.pool_ptrs[counter] = 1;
162 output.collapsed.push_back(first);
163 auto last = first;
164
165 for (Index_ i = 1; i < len; ++i) {
166 const auto& pp = collected[i];
167 auto current = pp.first;
168 if (current == last) {
169 output.reindex.pool_indices.push_back(pp.second);
170 ++(output.reindex.pool_ptrs[counter]);
171 continue;
172 }
173
174 Index_ pool_size = output.reindex.pool_indices.size();
175 counter = current - output.reindex.offset;
176 output.reindex.pool_ptrs[counter] = pool_size; // any overwrite is safe as the value is unchanged.
177 ++counter;
178 output.reindex.pool_indices.push_back(pp.second);
179 output.reindex.pool_ptrs[counter] = pool_size + 1;
180 output.collapsed.push_back(current);
181 last = current;
182 }
183 }
184
185 return output;
186}

210 void initialize(const Matrix<Value_, Index_>* mat, SparseParallelResults<Index_> processed, size_t extent, bool row, MaybeOracle<oracle_, Index_> oracle, Options opt) {
211 shift = extent - processed.collapsed.size();
212
213 needs_value = opt.sparse_extract_value;
214 needs_index = opt.sparse_extract_index;
215 needs_sort = opt.sparse_ordered_index;
216
217 if (needs_sort && needs_value) {
218 sortspace.reserve(extent);
219 }
220
221 // We need to extract indices for sorting and expansion purposes, even
222 // if they weren't actually requested.
223 opt.sparse_extract_index = true;
224 if (!needs_index) {
225 iholding.resize(processed.collapsed.size());
226 }
227
228 internal = new_extractor<true, oracle_>(mat, row, std::move(oracle), std::move(processed.collapsed), opt);
229 reindex = std::move(processed.reindex);

232public:
233 SparseRange<Value_, Index_> fetch(Index_ i, Value_* vbuffer, Index_* ibuffer) {
234 auto vinit = (needs_value ? vbuffer + shift : NULL);
235 auto iinit = (needs_index ? ibuffer + shift : iholding.data());
236 auto input = internal->fetch(i, vinit, iinit);
237
238 if (!needs_sort) {
239 // Pointers in 'input' and the two 'buffer' pointers may optionally point
240 // to overlapping arrays as long as each 'buffer' pointer precedes its
241 // corresponding pointer in 'input'. The idea is that the expansion of
242 // values into, e.g., 'vbuffer' will cause it to catch up to 'input.value'
243 // without clobbering any values in the latter. This assumes that
244 // 'input.value' has been shifted enough to make space for expansion; the
245 // required shift depends on the number of duplicates.
246 Index_ count = 0;
247 auto vcopy = vbuffer;
248 auto icopy = ibuffer;
249
250 auto vsrc = input.value;
251 bool replace_value = needs_value && vsrc != vcopy;
252
253 for (Index_ i = 0; i < input.number; ++i) {
254 auto lookup = input.index[i] - reindex.offset;
255 auto start = reindex.pool_ptrs[lookup];
256 auto num = reindex.pool_ptrs[lookup + 1] - start;
257 count += num;
258
259 if (replace_value) {
260 auto val = *vsrc; // make a copy just in case 'vcopy' and 'input.value' overlap.
261 std::fill_n(vcopy, num, val);
262 vcopy += num;
263 ++vsrc;
264 replace_value = (vcopy != vsrc); // if we've caught up, there no need to do this replacement.
265 }
266
267 if (needs_index) {
268 // Again, 'icopy' will eventually catch up to 'input.index' if
269 // they point to overlapping arrays. But we still need to
270 // replace values once we've managed to catch up, so we can't
271 // short-circuit like we did with 'replace_value'.
272 std::copy_n(reindex.pool_indices.begin() + start, num, icopy);
273 icopy += num;
274 }
275 }
276
277 input.number = count;
278 if (needs_value) {
279 input.value = vbuffer;
280 }
281 if (needs_index) {
282 input.index = ibuffer;
283 } else {
284 input.index = NULL;
285 }
286
287 } else if (needs_value) {
288 // This does not require any careful consideration of the overlaps
289 // between 'input' and 'buffers', as we're copying things into
290 // 'sortspace' anyway before copying them back into 'buffer'.
291 sortspace.clear();
292 for (Index_ i = 0; i < input.number; ++i) {
293 auto val = input.value[i];
294 auto lookup = input.index[i] - reindex.offset;
295 auto start = reindex.pool_ptrs[lookup];
296 auto end = reindex.pool_ptrs[lookup + 1];
297 for (Index_ j = start; j < end; ++j) {
298 sortspace.emplace_back(reindex.pool_indices[j], val);
299 }
300 }
301 std::sort(sortspace.begin(), sortspace.end());
302 input.number = sortspace.size();
303
304 auto vcopy = vbuffer;
305 for (const auto& ss : sortspace) {
306 *vcopy = ss.second;
307 ++vcopy;
308 }
309 input.value = vbuffer;
310
311 if (needs_index) {
312 auto icopy = ibuffer;
313 for (const auto& ss : sortspace) {
314 *icopy = ss.first;
315 ++icopy;
316 }
317 input.index = ibuffer;
318 } else {
319 input.index = NULL;
320 }
321
322 } else {
323 // Again, 'input.index' and 'ibuffer' may point to overlapping arrays,
324 // as long as the latter precedes the former; expansion into the latter
325 // will allow it to catch up to the former without clobbering, assuming
326 // that the latter was shifted back to provide enough space.
327 Index_ count = 0;
328 auto icopy = ibuffer;
329
330 for (Index_ i = 0; i < input.number; ++i) {
331 auto lookup = input.index[i] - reindex.offset;
332 auto start = reindex.pool_ptrs[lookup];
333 auto num = reindex.pool_ptrs[lookup + 1] - start;
334 count += num;
335
336 if (needs_index) {
337 std::copy_n(reindex.pool_indices.begin() + start, num, icopy);
338 icopy += num;
339 }
340 }
341
342 input.number = count;
343 if (needs_index) {
344 std::sort(ibuffer, ibuffer + count);
345 input.index = ibuffer;
346 } else {
347 input.index = NULL;
348 }
349 }
350
351 return input;
352 }
