260 if (reverse_indices.empty()) {
261 // Has duplicates or is out-of-order... need to expand the sparse vector into an array for indexing.
262 const T* ptr = NULL;
263 if constexpr(ROW) {
264 ptr = mat->row(r, inner_out_values, min_index, max_index, work);
265 } else {
266 ptr = mat->column(r, inner_out_values, min_index, max_index, work);
267 }
268
269 auto copy = out_indices;
270 for (size_t i = start; i < end; ++i) {
271 auto val = ptr[indices[i] - min_index];
272 if (val) { // note that this means that any deliberately inserted zero elements are lost.
273 *out_values = val;
274 ++out_values;
275 *out_indices = i;
276 ++out_indices;
277 }
278 }
279
280 return static_cast<size_t>(out_indices - copy);
281
282 } else {
283 // No duplicates, this just involves doing the reverse look-up.
284 SparseRange<T, IDX> range;
285 if constexpr(ROW) {
286 range = mat->sparse_row(r, inner_out_values, inner_out_indices, min_index, max_index, work, sorted);
287 } else {
288 range = mat->sparse_column(r, inner_out_values, inner_out_indices, min_index, max_index, work, sorted);
289 }
290
291 auto copy = out_indices;
292 for (size_t i = 0; i < range.number; ++i, ++range.index, ++range.value) {
293 auto final_idx = reverse_indices[*range.index];
294 if (final_idx != static_cast<IDX>(indices.size())) {
295 *out_values = *range.value;
296 ++out_values;
297 *out_indices = final_idx;
298 ++out_indices;
299 }
300 }
