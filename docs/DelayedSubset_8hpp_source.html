34 static void finish_assembly(
35 const std::vector<std::pair<Index_, Index_> >& collected,
36 const IndexStorage_& indices,
37 std::vector<Index_>& reverse_mapping,
38 std::vector<Index_>& unique_and_sorted,
39 Index_ mapping_dim,
40 std::vector<std::pair<Index_, Index_> >& mapping_duplicates,
41 std::vector<Index_>& mapping_duplicates_pool
42 ) {
43 unique_and_sorted.reserve(indices.size());
44 reverse_mapping.resize(indices.size());
45
46 mapping_duplicates.resize(mapping_dim);
47 mapping_duplicates_pool.reserve(indices.size());
48
49 for (Index_ i = 0, end = collected.size(); i < end; ++i) {
50 const auto& current = collected[i];
51 auto& range = mapping_duplicates[current.first];
52 if (unique_and_sorted.empty() || current.first != unique_and_sorted.back()) {
53 unique_and_sorted.push_back(current.first);
54 range.first = mapping_duplicates_pool.size();
55 }
56
57 mapping_duplicates_pool.push_back(current.second);
58 reverse_mapping[current.second] = unique_and_sorted.size() - 1;
59 ++range.second;
60 }
61 }

170 struct SparseBase {
171 /*
172 * The behavior of the temp buffers depends on whether we want sorted output.
173 *
174 * If we want sorted output, then the logic roughly follows that of DelayedSubsetUnique::SparseBase.
175 * We can directly extract into the user-supplied ibuffer/vbuffer and then copy everything into sortspace for resorting.
176 * The only exception is if the doesn't want the indices, in which case we need to do some allocation.
177 *
178 * If we don't want sorted output, then the logic follows that of DelayedSubsetSorted::SparseBase.
179 * Here, we can't extract into the user-supplied ibuffer/vbuffer, because we need some
180 * place to hold the to-be-processed values while we expand the duplicates.
181 * Of course, we can omit the vtemp allocation if the user doesn't want the values.
182 */
183
184 SparseBase(const Options& opt, size_t bufsize) : report_index(opt.sparse_extract_index), needs_sort(opt.sparse_ordered_index) {
185 if (needs_sort) {
186 if (!opt.sparse_extract_index) {
187 itemp.resize(bufsize);
188 }
189 sortspace.reserve(bufsize);
190 } else {
191 if (opt.sparse_extract_value) {
192 vtemp.resize(bufsize);
193 }
194 itemp.resize(bufsize);
195 }
196 }
197
198 protected:
199 bool report_index;
200 bool needs_sort;
201 std::vector<Value_> vtemp;
202 std::vector<Index_> itemp;
203 std::vector<std::pair<Index_, Value_> > sortspace;
204 };

208 std::unique_ptr<Extractor<DimensionSelectionType::INDEX, sparse_, Value_, Index_> > create_inner_extractor(const Options& opt, std::vector<Index_> idx) const {
209 if constexpr(sparse_) {
210 if (opt.sparse_ordered_index || !opt.sparse_extract_index) {
211 auto copy = opt;
212
213 // Turning off the sorting to enable possible optimizations in the underlying matrix.
214 // We don't need sorted output as we'll be resorting ourselves later.
215 copy.sparse_ordered_index = false;
216
217 // Need to force extraction of indices for correct duplication.
218 copy.sparse_extract_index = true;
219
220 return new_extractor<margin_ != 0, sparse_>(mat.get(), std::move(idx), copy);
221 }
222 }
223
224 return new_extractor<margin_ != 0, sparse_>(mat.get(), std::move(idx), opt);

227 template<class Extractor_>
228 static SparseRange<Value_, Index_> reorganize_sparse_unsorted(
229 Index_ i,
230 Value_* vbuffer,
231 Index_* ibuffer,
232 std::vector<Value_>& vtemp,
233 std::vector<Index_>& itemp,
234 bool report_index,
235 Extractor_* work,
236 const std::vector<std::pair<Index_, Index_> >& dups,
237 const std::vector<Index_>& pool)
238 {
239 // Allocation status of vtemp depends on the extraction mode used to construct 'work'.
240 Value_* vin = vtemp.data();
241
242 // itemp should always be allocated, as we need this to get the expanded counts.
243 Index_* iin = itemp.data();
244
245 auto raw = work->fetch(i, vin, iin);
246 if (!raw.value) {
247 vbuffer = NULL;
248 }
249 if (!report_index) {
250 ibuffer = NULL;
251 }
252
253 auto vcopy = vbuffer;
254 auto icopy = ibuffer;
255 Index_ counter = 0;
256
257 for (Index_ i = 0; i < raw.number; ++i) {
258 const auto& pool_pos = dups[raw.index[i]];
259 counter += pool_pos.second;
260
261 if (vcopy) {
262 std::fill(vcopy, vcopy + pool_pos.second, raw.value[i]);
263 vcopy += pool_pos.second;
264 }
265
266 if (icopy) {
267 auto istart = pool.begin() + pool_pos.first;
268 std::copy(istart, istart + pool_pos.second, icopy);
269 icopy += pool_pos.second;
270 }
271 }
272
273 return SparseRange<Value_, Index_>(counter, vbuffer, ibuffer);
274 }

276 template<class Extractor_>
277 static SparseRange<Value_, Index_> reorganize_sparse_sorted(
278 Index_ i,
279 Value_* vbuffer,
280 Index_* ibuffer,
281 std::vector<std::pair<Index_, Value_> >& sortspace,
282 std::vector<Index_>& itemp,
283 bool report_index,
284 Extractor_* work,
285 const std::vector<std::pair<Index_, Index_> >& dups,
286 const std::vector<Index_>& pool)
287 {
288 // itemp may not be allocated, if indices are to be extracted into ibuffer.
289 Index_* iin = (itemp.empty() ? ibuffer : itemp.data());
290 auto raw = work->fetch(i, vbuffer, iin);
291
292 sortspace.clear();
293 for (Index_ i = 0; i < raw.number; ++i) {
294 const auto& pool_pos = dups[raw.index[i]];
295 Index_ pool_end = pool_pos.first + pool_pos.second;
296
297 if (raw.value) {
298 for (Index_ j = pool_pos.first; j < pool_end; ++j) {
299 sortspace.emplace_back(pool[j], raw.value[i]);
300 }
301 } else {
302 for (Index_ j = pool_pos.first; j < pool_end; ++j) {
303 sortspace.emplace_back(pool[j], 0);
304 }
305 }
306 }
307
308 std::sort(sortspace.begin(), sortspace.end());
309
310 if (report_index) {
311 auto icopy = ibuffer;
312 for (const auto& x : sortspace) {
313 *icopy= x.first;
314 ++icopy;
315 }
316 } else {
317 ibuffer = NULL;
318 }
319
320 if (raw.value) {
321 auto vcopy = vbuffer;
322 for (const auto& x : sortspace) {
323 *vcopy = x.second;
324 ++vcopy;
325 }
326 } else {
327 vbuffer = NULL;
328 }
329
330 return SparseRange<Value_, Index_>(sortspace.size(), vbuffer, ibuffer);
331 }

388 void transplant_indices(
389 std::vector<Index_>& local,
390 std::vector<std::pair<Index_, Index_> >& collected,
391 std::vector<Index_>& reverse_mapping)
392 const {
393 std::sort(collected.begin(), collected.end());
394
395 reverse_mapping.resize(collected.size());
396 local.reserve(collected.size());
397
398 for (const auto& current : collected) {
399 if (local.empty() || current.first != local.back()) {
400 local.push_back(current.first);
401 }
402 reverse_mapping[current.second] = local.size() - 1;
403 }
404 }

406 void transplant_indices(
407 std::vector<Index_>& local,
408 std::vector<std::pair<Index_, Index_> >& collected,
409 std::vector<std::pair<Index_, Index_> >& dups,
410 std::vector<Index_>& pool)
411 const {
412 std::sort(collected.begin(), collected.end());
413
414 Index_ mapping_dim = margin_ == 0 ? mat->nrow() : mat->ncol();
415 dups.resize(mapping_dim);
416 pool.reserve(collected.size());
417 local.reserve(collected.size());
418
419 for (const auto& current : collected) {
420 auto& range = dups[current.first];
421 if (local.empty() || current.first != local.back()) {
422 local.push_back(current.first);
423 range.first = pool.size();
424 }
425 ++range.second;
426 pool.push_back(current.second);

433 BlockParallelExtractor(const DelayedSubset* parent, const Options& opt, Index_ bs, Index_ bl) {
434 this->block_start = bs;

437 const auto& parent_indices = parent->indices;
