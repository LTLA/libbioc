23void dimension_extremes(const Matrix<Value_, Index_>* p, int threads, StoreMinimum_& min_out, StoreMaximum_& max_out) {
24 auto dim = (row_ ? p->nrow() : p->ncol());
25 auto otherdim = (row_ ? p->ncol() : p->nrow());
26 const bool direct = p->prefer_rows() == row_;
27
28 constexpr bool store_min = !std::is_same<StoreMinimum_, bool>::value;
29 constexpr bool store_max = !std::is_same<StoreMaximum_, bool>::value;
30
31 if (!otherdim) {
32 return;
33 }
34
35 if (p->sparse()) {
36 Options opt;
37 opt.sparse_ordered_index = false;
38
39 if (direct) {
40 opt.sparse_extract_index = false;
41 parallelize([&](size_t, Index_ s, Index_ l) {
42 auto ext = consecutive_extractor<row_, true>(p, s, l, opt);
43 std::vector<Value_> vbuffer(otherdim);
44
45 for (Index_ i = s, e = s + l; i < e; ++i) {
46 auto out = ext->fetch(i, vbuffer.data(), NULL);
47
48 // If there's no non-zero values, we just leave the min/max as zero.
49 if (out.number) {
50 if constexpr(store_min) {
51 auto minned = *std::min_element(out.value, out.value + out.number);
52 if (minned < 0 || out.number == otherdim) {
53 min_out[i] = minned;
54 }
55 }
56 if constexpr(store_max) {
57 auto maxed = *std::max_element(out.value, out.value + out.number);
58 if (maxed > 0 || out.number == otherdim) {
59 max_out[i] = maxed;
60 }
61 }
62 }
63 }
64 }, dim, threads);
65
66 } else {
67 parallelize([&](size_t, Index_ s, Index_ l) {
68 auto ext = consecutive_extractor<!row_, true>(p, 0, otherdim, s, l, opt);
69 auto len = ext->block_length;
70 std::vector<Value_> vbuffer(len);
71 std::vector<Index_> ibuffer(len);
72 std::vector<Index_> counter(len);
73
74 for (Index_ i = 0; i < otherdim; ++i) {
75 auto out = ext->fetch(i, vbuffer.data(), ibuffer.data());
76 for (Index_ j = 0; j < out.number; ++j) {
77 auto idx = out.index[j];
78 auto val = static_cast<Output_>(out.value[j]);
79 if constexpr(store_min) {
80 auto& last = min_out[idx];
81 if (i == 0 || last > val) {
82 last = val;
83 }
84 }
85 if constexpr(store_max) {
86 auto& last = max_out[idx];
87 if (i == 0 || last < val) {
88 last = val;
89 }
90 }
91 ++counter[idx - s];
92 }
93 }
94
95 // Handling the zeros.
96 for (Index_ i = s, e = s + l; i < e; ++i) {
97 if (counter[i - s] < otherdim) {
98 if constexpr(store_min) {
99 if (min_out[i] > 0) {
100 min_out[i] = 0;
101 }
102 }
103 if constexpr(store_max) {
104 if (max_out[i] < 0) {
105 max_out[i] = 0;
106 }
107 }
108 }
109 }
110 }, dim, threads);
111 }
112
113 } else {
114 if (direct) {
115 parallelize([&](size_t, Index_ s, Index_ l) {
116 auto ext = consecutive_extractor<row_, false>(p, s, l);
117 std::vector<Value_> buffer(otherdim);
118 for (Index_ i = s, e = s + l; i < e; ++i) {
119 auto ptr = ext->fetch(i, buffer.data());
120 if constexpr(store_min) {
121 min_out[i] = *std::min_element(ptr, ptr + otherdim);
122 }
123 if constexpr(store_max) {
124 max_out[i] = *std::max_element(ptr, ptr + otherdim);
125 }
126 }
127 }, dim, threads);
128
129 } else {
130 parallelize([&](size_t, Index_ s, Index_ l) {
131 auto ext = consecutive_extractor<!row_, false>(p, 0, otherdim, s, l);
132 auto len = ext->block_length;
133 std::vector<Value_> buffer(len);
134
135 for (Index_ i = 0; i < otherdim; ++i) {
136 auto ptr = ext->fetch(i, buffer.data());
137 if (i) {
138 for (Index_ d = 0; d < len; ++d) {
139 auto idx = d + s;
140 auto val = static_cast<Output_>(ptr[d]);
141 if constexpr(store_min) {
142 auto& last = min_out[idx];
143 if (last > val) {
144 last = val;
145 }
146 }
147 if constexpr(store_max) {
148 auto& last = max_out[idx];
149 if (last < val) {
150 last = val;
151 }
152 }
153 }
154 } else {
155 if constexpr(store_min) {
156 std::copy(ptr, ptr + len, min_out + s);
157 }
158 if constexpr(store_max) {
159 std::copy(ptr, ptr + len, max_out + s);
160 }
161 }
162 }
163 }, dim, threads);
164 }
