23void dimension_extremes(const Matrix<Value_, Index_>* p, int threads, StoreMinimum_& min_out, StoreMaximum_& max_out) {
24 auto dim = (row_ ? p->nrow() : p->ncol());
25 auto otherdim = (row_ ? p->ncol() : p->nrow());
26 const bool direct = p->prefer_rows() == row_;
27
28 constexpr bool store_min = !std::is_same<StoreMinimum_, bool>::value;
29 constexpr bool store_max = !std::is_same<StoreMaximum_, bool>::value;
30
31 if (!otherdim) {
32 if constexpr(store_min) {
33 std::fill(min_out, min_out + dim, 0);
34 }
35 if constexpr(store_max) {
36 std::fill(max_out, max_out + dim, 0);
37 }
38 return;
39 }
40
41 if (p->sparse()) {
42 Options opt;
43 opt.sparse_ordered_index = false;
44
45 if (direct) {
46 opt.sparse_extract_index = false;
47 parallelize([&](size_t, Index_ s, Index_ l) {
48 auto ext = consecutive_extractor<row_, true>(p, s, l, opt);
49 std::vector<Value_> vbuffer(otherdim);
50
51 for (Index_ i = s, e = s + l; i < e; ++i) {
52 auto out = ext->fetch(i, vbuffer.data(), NULL);
53
54 if (out.number) {
55 if constexpr(store_min) {
56 auto minned = *std::min_element(out.value, out.value + out.number);
57 if (minned > 0 && out.number != otherdim) {
58 minned = 0;
59 }
60 min_out[i] = minned;
61 }
62 if constexpr(store_max) {
63 auto maxed = *std::max_element(out.value, out.value + out.number);
64 if (maxed < 0 && out.number != otherdim) {
65 maxed = 0;
66 }
67 max_out[i] = maxed;
68 }
69 } else {
70 if constexpr(store_min) {
71 min_out[i] = 0;
72 }
73 if constexpr(store_max) {
74 max_out[i] = 0;
75 }
76 }
77 }
78 }, dim, threads);
79
80 } else {

82 auto ext = consecutive_extractor<!row_, true>(p, 0, otherdim, s, l, opt);
83 auto len = ext->block_length;
84 std::vector<Value_> vbuffer(len);
85 std::vector<Index_> ibuffer(len);

87

91 auto idx = out.index[j];
92 auto& c = counter[idx - s];
93 auto val = static_cast<Output_>(out.value[j]);
94 if constexpr(store_min) {
95 auto& last = min_out[idx];
96 if (c == 0 || last > val) {
97 last = val;
98 }
99 }
100 if constexpr(store_max) {
101 auto& last = max_out[idx];
102 if (c == 0 || last < val) {
103 last = val;
104 }
105 }
106 ++c;
107 }
108 }
109
110 // Handling the zeros.
111 for (Index_ i = s, e = s + l; i < e; ++i) {
112 auto c = counter[i - s];
113 if (c == 0) {
114 if constexpr(store_min) {
115 min_out[i] = 0;
116 }
117 if constexpr(store_max) {
118 max_out[i] = 0;
119 }
120 } else if (c < otherdim) {
121 if constexpr(store_min) {
122 if (min_out[i] > 0) {
123 min_out[i] = 0;
124 }
125 }
126 if constexpr(store_max) {
127 if (max_out[i] < 0) {
128 max_out[i] = 0;
129 }
130 }
131 }
132 }
133 }, dim, threads);
134 }
135
136 } else {
137 if (direct) {
138 parallelize([&](size_t, Index_ s, Index_ l) {
139 auto ext = consecutive_extractor<row_, false>(p, s, l);
140 std::vector<Value_> buffer(otherdim);
141 for (Index_ i = s, e = s + l; i < e; ++i) {
142 auto ptr = ext->fetch(i, buffer.data());
143 if constexpr(store_min) {
144 min_out[i] = *std::min_element(ptr, ptr + otherdim);
145 }
146 if constexpr(store_max) {
147 max_out[i] = *std::max_element(ptr, ptr + otherdim);
148 }
149 }
150 }, dim, threads);
151
152 } else {
153 parallelize([&](size_t, Index_ s, Index_ l) {
154 auto ext = consecutive_extractor<!row_, false>(p, 0, otherdim, s, l);
155 auto len = ext->block_length;
156 std::vector<Value_> buffer(len);
157
158 // We already have a otherdim > 0 check above.
159 auto ptr = ext->fetch(0, buffer.data());
160 if constexpr(store_min) {
161 std::copy(ptr, ptr + len, min_out + s);
162 }
163 if constexpr(store_max) {
164 std::copy(ptr, ptr + len, max_out + s);
165 }
166
167 for (Index_ i = 1; i < otherdim; ++i) {
168 auto ptr = ext->fetch(i, buffer.data());
169 for (Index_ d = 0; d < len; ++d) {
170 auto idx = d + s;
171 auto val = static_cast<Output_>(ptr[d]);
172 if constexpr(store_min) {
173 auto& last = min_out[idx];
174 if (last > val) {
175 last = val;
176 }
177 }
178 if constexpr(store_max) {
179 auto& last = max_out[idx];
180 if (last < val) {
181 last = val;
182 }
183 }
184 }
185 }
186 }, dim, threads);
