265 template<class Store_>
266 void primary_dimension(Index_ i, const Index_* subset, Index_ length, PrimaryWorkspace& work, Store_& store) const {
267 if (!length) {
268 return;
269 }
270
271 auto iIt = indices.begin() + indptrs[i], eIt = indices.begin() + indptrs[i + 1];
272
273 if (indices[0]) { // Only jumping ahead if the start is non-zero.
274 bool do_cache = !work.cached.empty();
275 if (do_cache) {
276 if (work.cached[i] != -1) { // retrieving the jump from cache, if we came here before.
277 iIt += work.cached[i];
278 } else {
279 auto iIt2 = std::lower_bound(iIt, eIt, subset[0]);
280 work.cached[i] = iIt2 - iIt;
281 iIt = iIt2;
282 }
283 } else {
284 iIt = std::lower_bound(iIt, eIt, subset[0]);
285 }
286 }
287
288 if (iIt == eIt) {
289 return;
290 }
291
292 for (Index_ counter = 0; counter < length; ++counter) {
293 auto current = subset[counter];
294
295 while (iIt != eIt && current > *iIt) {
296 ++iIt;
297 }
298 if (iIt == eIt) {
299 break;
300 }
301
302 if (current == *iIt) {
303 int count = 0;
304 do {
305 ++iIt;
306 ++count;
307 } while (iIt != eIt && current == *iIt);
308 store.add(current, count);
309 } else {
310 store.skip(current);
311 }
312 }
313
314 return;
315 }

439 SecondaryWorkspace(Index_ max_index, const IndexStorage_& idx, const PointerStorage_& idp, Index_ start, Index_ length) : current_indptrs(length), current_indices(length) {
440 /* Here, the general idea is to store a local copy of the actual
441 * row indices (for CSC matrices; column indices, for CSR matrices)
442 * so that we don't have to keep on doing cache-unfriendly look-ups
443 * for the indices based on the pointers that we do have. This assumes
444 * that the density is so low that updates to the local indices are
445 * rare relative to the number of comparisons to those same indices.
446 * Check out the `secondary_dimension()` function for how this is used.
447 */
448 auto idpIt = idp.begin() + start;
449 for (Index_ i = 0; i < length; ++i, ++idpIt) {
450 current_indptrs[i].pointer = *idpIt;
451 current_indices[i] = (*idpIt < *(idpIt + 1) ? idx[*idpIt] : max_index);
452 }
453 return;
454 }
455
456 SecondaryWorkspace(Index_ max_index, const IndexStorage_& idx, const PointerStorage_& idp) :
457 SecondaryWorkspace(max_index, idx, idp, static_cast<Index_>(0), idp.size() - 1) {}
458
459 SecondaryWorkspace(Index_ max_index, const IndexStorage_& idx, const PointerStorage_& idp, const Index_* subset, Index_ length) :
460 current_indptrs(length), current_indices(length)
461 {
462 for (Index_ i0 = 0; i0 < length; ++i0) {
463 auto i = subset[i0];
464 current_indptrs[i0].pointer = idp[i];
465 current_indices[i0] = (idp[i] < idp[i + 1] ? idx[idp[i]] : max_index);
466 }
467 return;
468 }

486 void update_secondary_position(typename SecondaryWorkspace::Position& curptr, indptr_type limit) const {
487 if (!curptr.scanned) {
488 auto curdex = indices[curptr.pointer];
489 curptr.count = 0;
490 auto copy = curptr.pointer;
491 do {
492 ++copy;
493 } while (copy < limit && indices[copy] == curdex);
494 curptr.count = copy - curptr.pointer;
495 curptr.scanned = true;
496 }

499 template<class Store_>
500 void secondary_dimension_above(Index_ secondary, Index_ primary, Index_ index_primary, SecondaryWorkspace& work, Store_& output) const {
501 auto& curdex = work.current_indices[index_primary];
502 auto& curptr = work.current_indptrs[index_primary];
503
504 // Remember that we already store the index corresponding to the current indptr.
505 // So, we only need to do more work if the request is greater than that index.
506 if (secondary > curdex) {
507 auto max_index = max_secondary_index();
508 auto limit = indptrs[primary + 1];
509 update_secondary_position(curptr, limit);
510 curptr.pointer += curptr.count;
511 curptr.scanned = false;
512
513 // Having a peek at the index of the next non-zero
514 // element; maybe we're lucky enough that the requested
515 // index is below this, as would be the case for
516 // consecutive or near-consecutive accesses.
517 if (curptr.pointer < limit) {
518 auto candidate = indices[curptr.pointer];
519 if (candidate >= secondary) {
520 curdex = candidate;
521
522 } else if (secondary + 1 == max_index) {
523 // Special case if the requested index is at the end of the matrix,
524 // in which case we can just jump there directly rather than
525 // doing an unnecessary binary search. We can assume that there is
526 // at least one non-zero element in this dimension element (otherwise
527 // 'secondary > curdex' would not have passed, as curdex would be
528 // set to 'maxindex'), so 'limit - 1' is a valid expression here.
529 auto below_limit = limit - 1;
530 if (indices[below_limit] == secondary) {
531 curdex = secondary;
532 curptr.pointer = below_limit;
533 while (curptr.pointer > indptrs[primary] && indices[curptr.pointer - 1] == curdex) {
534 --curptr.pointer;
535 }
536 curptr.count = limit - curptr.pointer;
537 curptr.scanned = true;
538 } else {
539 curptr.pointer = limit;
540 curdex = max_index;
541 }
542
543 } else {
544 // Otherwise we need to search indices above the existing position.
545 // If the scanned count is available, we use it to trim the search space.
546 size_t offset = curptr.pointer + (curptr.scanned ? curptr.count : 0);
547 curptr.pointer = std::lower_bound(indices.begin() + offset, indices.begin() + limit, secondary) - indices.begin();
548 curdex = (curptr.pointer < limit ? indices[curptr.pointer] : max_index);
549 }
550 } else {
551 curdex = max_index;
552 }
553 }
554
555 if (secondary == curdex) { // assuming secondary < max_index, of course.
556 update_secondary_position(curptr, indptrs[primary + 1]);
557 output.add(primary, curptr.count);
558 } else {
559 output.skip(primary);
560 }
561 }

563 template<class Store_>
564 void secondary_dimension_below(Index_ secondary, Index_ primary, Index_ index_primary, SecondaryWorkspace& work, Store_& output) const {
565 auto& curdex = work.current_indices[index_primary];
566 auto& curptr = work.current_indptrs[index_primary];
567
568 if (secondary < curdex) {
569 auto limit = indptrs[primary + 1];
570 if (indptrs[primary] < limit) {
571 if (secondary == 0) {
572 // Special case if the requested index is at the end of the matrix,
573 // in which case we can just jump there directly rather than
574 // doing an unnecessary binary search.
575 curptr.pointer = indptrs[primary];
576 curdex = indices[curptr.pointer];
577 } else {
578 // Otherwise, searching indices below the existing position.
579 curptr.pointer = std::lower_bound(indices.begin() + indptrs[primary], indices.begin() + curptr.pointer, secondary) - indices.begin();
580 curdex = (curptr.pointer < limit ? indices[curptr.pointer] : max_secondary_index());
581 }
582 curptr.scanned = false;
583 }
584 }
585
586 if (secondary == curdex) { // assuming secondary < max_index, of course.
587 update_secondary_position(curptr, indptrs[primary + 1]);
588 output.add(primary, curptr.count);
589 } else {
590 output.skip(primary);
591 }
