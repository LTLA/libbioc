33template <bool row_, typename StoredValue_, typename InputValue_, typename InputIndex_>
34void convert_to_dense(const Matrix<InputValue_, InputIndex_>* incoming, StoredValue_* store, int threads = 1) {
35 InputIndex_ NR = incoming->nrow();
36 InputIndex_ NC = incoming->ncol();
37 InputIndex_ primary = (row_ ? NR : NC);
38 InputIndex_ secondary = (row_ ? NC : NR);
39
40 if (row_ == incoming->prefer_rows()) {
41 constexpr bool same_type = std::is_same<InputValue_, StoredValue_>::value;
42 parallelize([&](size_t, InputIndex_ start, InputIndex_ length) -> void {
43 std::vector<InputValue_> temp(same_type ? 0 : secondary);
44 auto store_copy = store + multiply(start, secondary);
45 auto wrk = consecutive_extractor<false>(incoming, row_, start, length);
46
47 for (InputIndex_ x = 0; x < length; ++x) {
48 if constexpr(same_type) {
49 auto ptr = wrk->fetch(store_copy);
50 copy_n(ptr, secondary, store_copy);
51 } else {
52 auto ptr = wrk->fetch(temp.data());
53 std::copy_n(ptr, secondary, store_copy);
54 }
55 store_copy += secondary;
56 }
57 }, primary, threads);
58
59 } else {
60 // We iterate on the incoming matrix's preferred dimension, under the
61 // assumption that it may be arbitrarily costly to extract in the
62 // non-preferred dim; it is thus cheaper to do cache-unfriendly inserts
63 // into the output buffers.
64
65 parallelize([&](size_t, InputIndex_ start, InputIndex_ length) -> void {
66 auto wrk = consecutive_extractor<false>(incoming, !row_, 0, secondary, start, length);
67 std::vector<InputValue_> temp(length);
68 auto store_copy = store + multiply(start, secondary);
69
70 for (InputIndex_ x = 0; x < secondary; ++x) {
71 auto ptr = wrk->fetch(temp.data());
72 auto bptr = store_copy;
73 for (InputIndex_ p = 0; p < length; ++p, bptr += secondary) {
74 *bptr = ptr[p];
75 }
76 ++store_copy;
77 }
78 }, primary, threads);
79 }
80
81 return;
82}
