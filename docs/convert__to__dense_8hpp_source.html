31void convert_to_dense(const Matrix_* incoming, StoredValue_* store, int threads = 1) {
32 typedef typename Matrix_::index_type Index_;
33 typedef typename Matrix_::value_type Value_;
34
35 size_t NR = incoming->nrow();
36 size_t NC = incoming->ncol();
37 size_t primary = (row_ ? NR : NC);
38 size_t secondary = (row_ ? NC : NR);
39
40 if (row_ == incoming->prefer_rows()) {
41 constexpr bool same_type = std::is_same<Value_, StoredValue_>::value;
42 parallelize([&](size_t, Index_ start, Index_ length) -> void {
43 std::vector<Value_> temp(same_type ? 0 : secondary);
44 auto store_copy = store + start * secondary;
45 auto wrk = consecutive_extractor<row_, false, Value_, Index_>(incoming, start, length);
46
47 for (Index_ p = start, e = start + length; p < e; ++p, store_copy += secondary) {
48 if constexpr(same_type) {
49 wrk->fetch_copy(p, store_copy);
50 } else {
51 auto ptr = wrk->fetch(p, temp.data());
52 std::copy(ptr, ptr + secondary, store_copy);
53 }
54 }
55 }, primary, threads);
56
57 } else {
58 // We iterate on the incoming matrix's preferred dimension, under the
59 // assumption that it may be arbitrarily costly to extract in the
60 // non-preferred dim; it is thus cheaper to do cache-unfriendly inserts
61 // into the output buffers.
62
63 parallelize([&](size_t, Index_ start, Index_ length) -> void {
64 auto wrk = consecutive_extractor<!row_, false, Value_, Index_>(incoming, 0, secondary, start, length);
65 auto len = wrk->block_length;
66 std::vector<Value_> temp(len);
67 auto store_copy = store + start * secondary;
68
69 for (Index_ s = 0; s < secondary; ++s, ++store_copy) {
70 auto ptr = wrk->fetch(s, temp.data());
71 auto bptr = store_copy;
72 for (size_t p = 0; p < len; ++p, bptr += secondary) {
73 *bptr = ptr[p];
74 }
75 }
76 }, primary, threads);
77 }
