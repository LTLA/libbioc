61 void addline(size_t row, size_t col, size_t data, size_t) {
62 // Assigning each line to a block based on its integer size.
63 constexpr int max8 = std::numeric_limits<uint8_t>::max();
64 constexpr int max16 = std::numeric_limits<uint16_t>::max();
65
66 if (data > max16) {
67 category[row] = std::max(category[row], static_cast<uint8_t>(2));
68 } else if (data > max8) {
69 category[row] = std::max(category[row], static_cast<uint8_t>(1));
70 }
71
72 ++lines_per_row[row];
73 return;
74 }

79 template<typename B>
80 void add(const B* buffer, size_t n) {
81 base.add(buffer, n, *this);
82 return;
83 }
84
85 void finish() {
86 base.finish(*this);
87
88 // Computing the number of features and lines in each block.
89 auto iIt = index.begin();
90 auto cIt = lines_per_row.begin();
91 for (auto f : category) {
92 auto& current = rows_per_category[f];
93
94 (*iIt) = current;
95 ++current;
96 ++iIt;
97
98 lines_per_category[f] += *cIt;
99 ++cIt;
100 }
101
102 // Computing the permutation.
103 auto cumsum = rows_per_category;
104 size_t last = 0;
105 for (auto& x : cumsum) {
106 std::swap(x, last);
107 last += x;
108 }
109 for (size_t i = 0; i < nrows; ++i) {
110 permutation[i] = cumsum[category[i]] + index[i];
111 }
112
113 return;
114 }

233 LineAssignments ass;
234 process(ass);
235 ass.finish();
236
237 LayeredMatrixData<T, IDX> output;

243 process(builder);
244 output.matrix = builder.template finish<T, IDX>();
