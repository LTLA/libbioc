149 SparseRange<T, IDX> subset_sparse(size_t i, T* out_values, IDX* out_indices, InputWorkspace* work, bool sorted) const {
150 SparseRange<T, IDX> output;
151
152 if constexpr(WORKROW) {
153 output = mat->sparse_row(i, out_values, out_indices, work, sorted);
154 } else {
155 output = mat->sparse_column(i, out_values, out_indices, work, sorted);
156 }
157
158 if (block_start) {
159 if (out_indices != output.index) {
160 std::copy(output.index, output.index + output.number, out_indices);
161 output.index = out_indices;
162 }
163 for (size_t i = 0; i < output.number; ++i) {
164 out_indices[i] -= block_start;
165 }
166 }
167
168 return output;
