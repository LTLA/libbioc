255 template<bool accrow_, DimensionSelectionType selection_, bool sparse_, typename ... Args_>
256 std::unique_ptr<Extractor<selection_, sparse_, Value_, Index_> > populate(const Options& opt, Args_&& ... args) const {
257 std::unique_ptr<Extractor<selection_, sparse_, Value_, Index_> > output;
258
259 if constexpr(accrow_ != (margin_ == 0)) {
260 if constexpr(sparse_) {
261 output.reset(new SparseAlongExtractor<selection_>(this, opt, std::forward<Args_>(args)...));
262 } else {
263 output.reset(new DenseAlongExtractor<selection_>(this, opt, std::forward<Args_>(args)...));
264 }
265 } else {
266 auto ptr = new_extractor<accrow_, sparse_>(this->mat.get(), std::forward<Args_>(args)..., opt);
267 if constexpr(sparse_) {
268 output.reset(new SparseAcrossExtractor<selection_>(std::move(ptr), this->block_start));
269 } else {
270 output.reset(new DenseAcrossExtractor<selection_>(std::move(ptr), this->block_start));
271 }
272 }
273
274 return output;
275 }
