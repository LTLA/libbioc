424 FragmentedSparseMatrix(Index_ nr, Index_ nc, ValueVectorStorage_ vals, IndexVectorStorage_ idx, bool row, bool check = true) :
425 nrows(nr), ncols(nc), values(std::move(vals)), indices(std::move(idx)), row_based(row)
426 {
427 if (check) {
428 if (values.size() != indices.size()) {
429 throw std::runtime_error("'values' and 'indices' should be of the same length");
430 }
431
432 if (row_based) {
433 if (indices.size() != static_cast<size_t>(nrows)) {
434 throw std::runtime_error("length of 'indices' should be equal to number of rows'");
435 }
436 } else {
437 if (indices.size() != static_cast<size_t>(ncols)) {
438 throw std::runtime_error("length of 'indices' should be equal to number of columns");
439 }
440 }
441
442 ElementType<ElementType<IndexVectorStorage_> > max_index = (row_based ? ncols : nrows);
443 for (size_t i = 0, end = indices.size(); i < end; ++i) {
444 const auto& curv = values[i];
445 const auto& curi = indices[i];
446 if (curv.size() != curi.size()) {
447 throw std::runtime_error("corresponding elements of 'values' and 'indices' should have the same length");
448 }
449
450 for (auto x : curi) {
451 if (x < 0 || x >= max_index) {
452 throw std::runtime_error("'indices' should contain non-negative integers less than the number of " + (row_based ? std::string("columns") : std::string("rows")));
453 }
454 }
455
456 for (size_t j = 1, jend = curi.size(); j < jend; ++j) {
457 if (curi[j] <= curi[j - 1]) {
458 throw std::runtime_error("indices should be strictly increasing within each element of 'indices'");
459 }
460 }
461 }
462 }
