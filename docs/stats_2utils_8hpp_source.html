53template<bool parallel_ = true, class Function_, typename Index_>
54void parallelize(Function_ fun, Index_ tasks, size_t threads) {
55 if constexpr(parallel_) {
56 if (threads > 1) {
57#ifndef TATAMI_CUSTOM_PARALLEL
58 Index_ worker_size = (tasks / threads) + (tasks % threads > 0); // Ceiling of an integer division.
59 threads = (tasks / worker_size) + (tasks % worker_size > 0); // Set the actual number of required threads.
60 std::vector<std::string> errors(threads);
61
62#if defined(_OPENMP)
63 #pragma omp parallel for num_threads(threads)
64 for (size_t t = 0; t < threads; ++t) {
65 Index_ start = worker_size * t; // Will not overflow due to the above recomputation of 'threads'.
66 Index_ remaining = tasks - start; // Must be positive, as otherwise 'tasks % worker_size = 0' and the iteration wouldn't even get here.
67
68 try {
69 fun(t, start, std::min(remaining, worker_size)); // Use 'remaining' to avoid potential overflow from computing 'end = start + worker_size'.
70 } catch (std::exception& e) {
71 errors[t] = e.what();
72 } catch (...) {
73 errors[t] = "unknown error in thread " + std::to_string(t);
74 }
75 }
76
77#else
78 Index_ first = 0;
79 std::vector<std::thread> workers;
80 workers.reserve(threads);
81
82 for (size_t t = 0; t < threads && first < tasks; ++t) {
83 Index_ remaining = tasks - first;
84 Index_ len = std::min(remaining, worker_size);
85 workers.emplace_back([&fun,&errors](int t, Index_ first, Index_ len) -> void {
86 try {
87 fun(t, first, len);
88 } catch (std::exception& e) {
89 errors[t] = e.what();
90 } catch (...) {
91 errors[t] = "unknown error in thread " + std::to_string(t);
92 }
93 }, t, first, len);
94 first += len;
95 }
96
97 for (auto& wrk : workers) {
98 wrk.join();
99 }
100#endif
101
102 for (const auto& e : errors) {
103 if (!e.empty()) {
104 throw std::runtime_error(e);
105 }
106 }
107
108#else
109 TATAMI_CUSTOM_PARALLEL(std::move(fun), tasks, threads);
110#endif
111 return;
112 }
113 }
114
115 fun(0, 0, tasks);
116 return;
117}
