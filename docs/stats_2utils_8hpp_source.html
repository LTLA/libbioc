36template<bool parallel_ = true, class Function_>
37void parallelize(Function_ fun, size_t tasks, size_t threads) {
38#if defined(_OPENMP) || defined(TATAMI_CUSTOM_PARALLEL)
39 if constexpr(parallel_) {
40 size_t worker_size = std::ceil(static_cast<double>(tasks) / static_cast<double>(threads));
41
42 if (threads > 1) {
43#ifndef TATAMI_CUSTOM_PARALLEL
44 #pragma omp parallel for num_threads(threads)
45 for (size_t t = 0; t < threads; ++t) {
46 size_t start = worker_size * t, end = std::min(tasks, start + worker_size);
47 if (start < end) {
48 fun(t, start, end);
49 }
50 }
51#else
52 TATAMI_CUSTOM_PARALLEL(std::move(fun), tasks, threads, worker_size);
53#endif
54 return;
55 }
56 }
57#endif
58
59 fun(0, 0, tasks);
60 return;
61}
