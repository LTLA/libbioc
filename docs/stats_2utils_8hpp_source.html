53void parallelize(Function_ fun, Index_ tasks, size_t threads) {
54 if constexpr(parallel_) {
55 if (threads > 1) {
56#ifndef TATAMI_CUSTOM_PARALLEL
57 Index_ worker_size = (tasks / threads) + (tasks % threads > 0); // Ceiling of an integer division.
58 threads = (tasks / worker_size) + (tasks % worker_size > 0); // Set the actual number of required threads.
59 std::vector<std::string> errors(threads);
60
61#if defined(_OPENMP)
62 #pragma omp parallel for num_threads(threads)
63 for (size_t t = 0; t < threads; ++t) {
64 Index_ start = worker_size * t; // Will not overflow due to the above recomputation of 'threads'.
65 Index_ remaining = tasks - start; // Must be positive, as otherwise 'tasks % worker_size = 0' and the iteration wouldn't even get here.
66
67 try {
68 fun(t, start, std::min(remaining, worker_size)); // Use 'remaining' to avoid potential overflow from computing 'end = start + worker_size'.
69 } catch (std::exception& e) {
70 errors[t] = e.what();
71 } catch (...) {
72 errors[t] = "unknown error in thread " + std::to_string(t);
73 }
74 }
75
76#else
77 Index_ first = 0;
78 std::vector<std::thread> workers;
79 workers.reserve(threads);
80
81 for (size_t t = 0; t < threads && first < tasks; ++t) {
82 Index_ remaining = tasks - first;
83 Index_ len = std::min(remaining, worker_size);
84 workers.emplace_back([&fun,&errors](int t, Index_ first, Index_ len) -> void {
85 try {
86 fun(t, first, len);
87 } catch (std::exception& e) {
88 errors[t] = e.what();

94 }

97 wrk.join();
98 }
99#endif

102 if (!e.empty()) {
103 throw std::runtime_error(e);
104 }
105 }
106
107#else
108 TATAMI_CUSTOM_PARALLEL(std::move(fun), tasks, threads);

110 return;
111 }
112 }
