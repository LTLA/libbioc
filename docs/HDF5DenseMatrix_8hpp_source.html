109public:
119 HDF5DenseMatrix(std::string file, std::string name, size_t cache_limit = 100000000) :
120 file_name(std::move(file)),
121 dataset_name(std::move(name))
122 {
123 H5::H5File fhandle(file_name, H5F_ACC_RDONLY);
124 auto dhandle = fhandle.openDataSet(dataset_name);
125 auto space = dhandle.getSpace();
126
127 int ndim = space.getSimpleExtentNdims();
128 if (ndim != 2) {
129 throw std::runtime_error("data in HDF5 dataset is not a two-dimensional array");
130 }
131
132 hsize_t dims_out[2];
133 space.getSimpleExtentDims(dims_out, NULL);
134 firstdim = dims_out[0];
135 seconddim = dims_out[1];
136
137 auto curtype = dhandle.getTypeClass();
138 if (curtype != H5T_INTEGER && curtype != H5T_FLOAT) {
139 throw std::runtime_error("expected numeric data in the HDF5 dataset");
140 }
141
142 auto dparms = dhandle.getCreatePlist();
143 size_t num_chunks_per_firstdim, num_chunks_per_seconddim;
144
145 if (dparms.getLayout() != H5D_CHUNKED) {
146 // If contiguous, each firstdim is treated as a chunk.
147 // So the number of chunks along the seconddim
148 num_chunks_per_firstdim = 1;
149 num_chunks_per_seconddim = firstdim;
150 } else {
151 hsize_t chunk_dims[2];
152 dparms.getChunk(2, chunk_dims);
153 const size_t chunk_firstdim = chunk_dims[0];
154 const size_t chunk_seconddim = chunk_dims[1];
155
156 // Need to use the other dimension to figure out
157 // the number of chunks along one dimension.
158 num_chunks_per_firstdim = std::ceil(static_cast<double>(seconddim)/chunk_seconddim);
159 num_chunks_per_seconddim = std::ceil(static_cast<double>(firstdim)/chunk_firstdim);
160
161 size_t chunk_size = dhandle.getDataType().getSize() * chunk_firstdim * chunk_seconddim;
162 cache_size_firstdim = std::min(cache_limit, num_chunks_per_firstdim * chunk_size);
163 cache_size_seconddim = std::min(cache_limit, num_chunks_per_seconddim * chunk_size);
164
165 // We used to be able to compute the nslots exactly, but then we got:
166 // https://forum.hdfgroup.org/t/unintended-behaviour-for-hash-values-during-chunk-caching/4869/5
167 // ... so we'll just set the number of slots to the total number of chunks and forget about it.
168 nslots = num_chunks_per_firstdim * num_chunks_per_seconddim;
169 }
170
171 prefer_firstdim = num_chunks_per_firstdim < num_chunks_per_seconddim;
172 return;
173 }

208 HDF5DenseWorkspace(const std::string& fpath, const std::string& dpath, size_t nslots, size_t cache_size, size_t dim) :
209 current_size(dim),
210 memspace(1, &current_size)
211 {
212 // Configuring the chunk cache.
213 if (nslots) {
214 H5::FileAccPropList plist(H5::FileAccPropList::DEFAULT.getId());
215
216 /* The first argument is ignored, according to https://support.hdfgroup.org/HDF5/doc/RM/RM_H5P.html.
217 * Setting w0 to 0 to evict the last used chunk; no need to worry about full vs partial reads here.
218 */
219 plist.setCache(10000, nslots, cache_size, 0);
220
221 file.openFile(fpath, H5F_ACC_RDONLY, plist);
222 } else {
223 file.openFile(fpath, H5F_ACC_RDONLY);
224 }
225
226 dataset = file.openDataSet(dpath);
227 dataspace = dataset.getSpace();
228 dataspace.selectNone();
229 memspace.selectAll();
230 return;
231 }

264 template<bool row>
265 const T* extract(size_t i, T* buffer, size_t first, size_t last, HDF5DenseWorkspace& work) const {
266 if constexpr(row != transpose) {
267 work.offset[0] = i;
268 work.offset[1] = first;
269 work.count[0] = 1;
270 work.count[1] = last - first;
271 } else {
272 work.offset[0] = first;
273 work.offset[1] = i;
274 work.count[0] = last - first;
275 work.count[1] = 1;
276 }
277
278 work.dataspace.selectHyperslab(H5S_SELECT_SET, work.count, work.offset);
279
280 if (static_cast<size_t>(work.current_size) != last - first) {
281 work.current_size = last - first;
282 work.memspace.setExtentSimple(1, &(work.current_size));
283 work.memspace.selectAll();
284 }
285
286 work.dataset.read(buffer, HDF5::define_mem_type<T>(), work.memspace, work.dataspace);
287 return buffer;
288 }
