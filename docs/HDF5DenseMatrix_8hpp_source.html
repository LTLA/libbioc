117 HDF5DenseMatrix(std::string file, std::string name, size_t cache_limit = 100000000) :
118 file_name(std::move(file)),
119 dataset_name(std::move(name))
120 {
121 H5::H5File fhandle(file_name, H5F_ACC_RDONLY);
122 auto dhandle = fhandle.openDataSet(dataset_name);
123 auto space = dhandle.getSpace();
124
125 int ndim = space.getSimpleExtentNdims();
126 if (ndim != 2) {
127 throw std::runtime_error("data in HDF5 dataset is not a two-dimensional array");
128 }
129
130 hsize_t dims_out[2];
131 space.getSimpleExtentDims(dims_out, NULL);
132 firstdim = dims_out[0];
133 seconddim = dims_out[1];
134
135 auto curtype = dhandle.getTypeClass();
136 if (curtype != H5T_INTEGER && curtype != H5T_FLOAT) {
137 throw std::runtime_error("expected numeric data in the HDF5 dataset");
138 }
139
140 auto dparms = dhandle.getCreatePlist();
141 if (dparms.getLayout() != H5D_CHUNKED) {
142 // If contiguous, each firstdim is treated as a chunk.
143 chunk_firstdim = 1;
144 chunk_seconddim = seconddim;
145 } else {
146 hsize_t chunk_dims[2];
147 dparms.getChunk(2, chunk_dims);
148 chunk_firstdim = chunk_dims[0];
149 chunk_seconddim = chunk_dims[1];
150 }
151
152 auto limit_cache = [](size_t chunk_dim, size_t mydim, size_t otherdim, size_t type_size, size_t limit) -> size_t {
153 size_t nelements_along_chunkdim = std::min(mydim, limit / (type_size * otherdim));
154 size_t nchunks = nelements_along_chunkdim / chunk_dim;
155 return std::min(nelements_along_chunkdim, nchunks * chunk_dim);
156 };
157
158 size_t data_size = dhandle.getDataType().getSize();
159 cache_firstdim = limit_cache(chunk_firstdim, firstdim, seconddim, data_size, cache_limit);
160 cache_seconddim = limit_cache(chunk_seconddim, seconddim, firstdim, data_size, cache_limit);
161
162 // Favoring extraction along the first dimension if it's not crippled
163 // by the chunking. This favors pulling of contiguous sections along
164 // the last (i.e., second) dimension, which is how HDF5 stores and
165 // returns 2D arrays, so we avoid needing some manual transposition.
166 if (cache_firstdim >= chunk_firstdim) {
167 prefer_firstdim = true;
168 } else if (cache_seconddim >= chunk_seconddim) {
169 // Of course, if the first dimension is chunk-compromised, then we
170 // would rather do the transpositions than repeat a read from file.
171 prefer_firstdim = false;
172 } else {
173 // If both are crippled, then we might as well extract along the first dimension.
174 prefer_firstdim = true;
175 }
176
177 return;
178 }

268 /* We manually handle the chunk caching, in particular so that we only
269 * pay the cost of rearranging the chunked data into contiguous arrays once.
270 */
271 template<bool row>
272 const T* extract(size_t i, T* buffer, size_t first, size_t last, HDF5DenseWorkspace& work) const {
273 // Figuring out which chunk the request belongs to.
274 hsize_t cache_mydim, chunk_otherdim, mydim, otherdim;
275 if constexpr(row != transpose) {
276 cache_mydim = cache_firstdim;
277 chunk_otherdim = chunk_seconddim;
278 mydim = firstdim;
279 otherdim = seconddim;
280 } else {
281 cache_mydim = cache_seconddim;
282 chunk_otherdim = chunk_firstdim;
283 mydim = seconddim;
284 otherdim = firstdim;
285 }
286
287 // No caching can be done here, so we just extract directly.
288 if (cache_mydim == 0) {
289 return extract<row>(i, buffer, first, last, work.dataset, work.dataspace, work.memspace);
290 }
291
292 size_t chunk = i / cache_mydim;
293 size_t index = i % cache_mydim;
294
295 // If we're in the same chunk for 'i' and the 'first'/'last' are within
296 // range, we re-use the cache. In theory, we could minimize
297 // extractions if the 'first' and 'last' are overlapping with their
298 // cached counterparts, but it's pretty rare that 'first' and 'last'
299 // will change, so we'll just keep things simple.
300 if (chunk != work.cached_chunk || first < work.cached_first || last > work.cached_last) {
301 size_t first_chunk = first / chunk_otherdim;
302
303 // Need the chunk _after_ the chunk containing the _before-last_ element.
304 size_t last_chunk = last ? (last - 1) / chunk_otherdim + 1 : 0;
305
306 hsize_t new_cached_first = first_chunk * chunk_otherdim;
307 hsize_t new_cached_last = std::min(otherdim, last_chunk * chunk_otherdim);
308 size_t new_span = new_cached_last - new_cached_first;
309
310 hsize_t cache_mydim_start = chunk * cache_mydim;
311 hsize_t cache_mydim_end = std::min(mydim, cache_mydim_start + cache_mydim);
312 hsize_t cache_mydim_actual = cache_mydim_end - cache_mydim_start;
313 size_t new_cache_size = new_span * cache_mydim;
314
315 T* destination;
316 work.cache.resize(new_cache_size);
317 if constexpr(row != transpose) {
318 destination = work.cache.data();
319 } else {
320 work.buffer.resize(new_cache_size);
321 destination = work.buffer.data();
322 }
323
324 // For chunks not present in the cache, we need to loop through and read them in.
325 hsize_t offset[2];
326 hsize_t count[2];
327 {
328 constexpr int x = (row != transpose);
329 offset[1-x] = cache_mydim_start;
330 offset[x] = new_cached_first;
331 count[1-x] = cache_mydim_actual;
332 count[x] = new_span;
333 }
334 work.dataspace.selectHyperslab(H5S_SELECT_SET, count, offset);
335
336 // HDF5 is a lot faster when the memspace and dataspace match in dimensionality.
337 // Presumably there is some shuffling that happens inside when dimensions don't match.
338 work.memspace.setExtentSimple(2, count);
339 work.memspace.selectAll();
340
341 work.dataset.read(destination, HDF5::define_mem_type<T>(), work.memspace, work.dataspace);
342 if constexpr(row == transpose) {
343 auto output = work.cache.begin();
344 for (hsize_t x = 0; x < count[1]; ++x, output += new_span) {
345 auto in = work.buffer.begin() + x;
346 for (hsize_t y = 0; y < count[0]; ++y, in += count[1]) {
347 *(output + y) = *in;
348 }
349 }
350 }
351
352 work.cached_first = new_cached_first;
353 work.cached_last = new_cached_last;
354 work.cached_chunk = chunk;
355 }
356
357 auto start = work.cache.begin() + index * (work.cached_last - work.cached_first);
358 std::copy(
359 start + (first - work.cached_first),
360 start + (last - work.cached_first),
361 buffer
362 );
363 return buffer;
364 }
