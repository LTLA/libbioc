232 /* We manually handle the chunk caching, in particular so that we only

235 template<bool row>
236 const T* extract(size_t i, T* buffer, size_t first, size_t last, HDF5DenseWorkspace& work) const {
237 // Figuring out which chunk the request belongs to.
238 hsize_t cache_mydim, chunk_otherdim, mydim, otherdim;
239 if constexpr(row != transpose) {
240 cache_mydim = cache_firstdim;
241 chunk_otherdim = chunk_seconddim;
242 mydim = firstdim;
243 otherdim = seconddim;
244 } else {
245 cache_mydim = cache_seconddim;
246 chunk_otherdim = chunk_firstdim;
247 mydim = seconddim;
248 otherdim = firstdim;
249 }
250
251 // No caching can be done here, so we just extract directly.
252 if (cache_mydim == 0) {

256 {
257#else

259#endif

262
263#ifndef TATAMI_HDF5_PARALLEL_LOCK
264 }
265#else
266 });
267#endif
268 return out;
269 }

271 size_t chunk = i / cache_mydim;
272 size_t index = i % cache_mydim;
273
274 // If we're in the same chunk for 'i' and the 'first'/'last' are within
275 // range, we re-use the cache. In theory, we could minimize
276 // extractions if the 'first' and 'last' are overlapping with their
277 // cached counterparts, but it's pretty rare that 'first' and 'last'
278 // will change, so we'll just keep things simple.
279 if (chunk != work.cached_chunk || first < work.cached_first || last > work.cached_last) {
280 size_t first_chunk = first / chunk_otherdim;
281
282 // Need the chunk _after_ the chunk containing the _before-last_ element.
283 size_t last_chunk = last ? (last - 1) / chunk_otherdim + 1 : 0;
284
285 hsize_t new_cached_first = first_chunk * chunk_otherdim;
286 hsize_t new_cached_last = std::min(otherdim, last_chunk * chunk_otherdim);
287 size_t new_span = new_cached_last - new_cached_first;
288
289 hsize_t cache_mydim_start = chunk * cache_mydim;
290 hsize_t cache_mydim_end = std::min(mydim, cache_mydim_start + cache_mydim);
291 hsize_t cache_mydim_actual = cache_mydim_end - cache_mydim_start;
292 size_t new_cache_size = new_span * cache_mydim;
293
294 T* destination;
295 work.cache.resize(new_cache_size);
296 if constexpr(row != transpose) {
297 destination = work.cache.data();
298 } else {
299 work.buffer.resize(new_cache_size);
300 destination = work.buffer.data();
301 }
302
303 // For chunks not present in the cache, we need to loop through and read them in.
304 hsize_t offset[2];
305 hsize_t count[2];
306 {
307 constexpr int x = (row != transpose);
308 offset[1-x] = cache_mydim_start;
309 offset[x] = new_cached_first;
310 count[1-x] = cache_mydim_actual;
311 count[x] = new_span;
312 }
