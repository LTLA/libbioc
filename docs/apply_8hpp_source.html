382template<int MARGIN, typename T, typename IDX, class Factory>
383void apply(const Matrix<T, IDX>* p, Factory& factory, int threads = 1) {
384 size_t NR = p->nrow(), NC = p->ncol();
385
386 /* One might question why we use MARGIN in the template if we just convert
387 * it to ROW here. This is because 'MARGIN' is used when we're doing
388 * something to the matrix; 'ROW' is used for the representation of the
389 * matrix itself. I'm keeping the distinction clear in the interface.
390 */
391 constexpr bool ROW = MARGIN == 0;
392
393 const size_t dim = (ROW ? NR : NC);
394 const size_t otherdim = (ROW ? NC : NR);
395
396 /* If we support running calculations AND the preference
397 * is not consistent with the margin, we give it a shot.
398 */
399 if constexpr(stats::has_sparse_running<Factory>::value || stats::has_dense_running<Factory>::value) {
400 if (p->prefer_rows() != ROW){
401 if constexpr(stats::has_sparse_running<Factory>::value) {
402 if (p->sparse()) {
403 stats::apply_sparse_running<MARGIN == 0>(dim, otherdim, p, factory, threads);
404 return;
405 }
406 }
407
408 stats::apply_dense_running<MARGIN == 0>(dim, otherdim, p, factory, threads);
409 return;
410 }
411 }
412
413 if constexpr(stats::has_sparse_direct<Factory>::value) {
414 if (p->sparse()) {
415 stats::apply_sparse_direct<MARGIN == 0>(dim, otherdim, p, factory, threads);
416 return;
417 }
418 }
419
420 stats::apply_dense_direct<MARGIN == 0>(dim, otherdim, p, factory, threads);
421 return;
422}
