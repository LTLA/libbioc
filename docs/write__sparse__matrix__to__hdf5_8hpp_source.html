181 // Firstly, figuring out the number of non-zero elements, and the maximum value of the indices and data.
182 T lower_data = 0, upper_data = 0;
183 IDX upper_index = 0;
184 size_t NR = mat->nrow(), NC = mat->ncol();
185 hsize_t non_zeros = 0;
186 bool non_integer = false;
187
188 auto update_stats = [&](const SparseRange<T, IDX>& extracted) -> void {
189 non_zeros += extracted.number;
190 for (size_t i = 0; i < extracted.number; ++i) {
191 auto val = extracted.value[i];
192 if constexpr(!std::is_integral<T>::value) {
193 if (std::trunc(val) != val || std::isinf(val)) {
194 non_integer = true;
195 }
196 }
197
198 if (val < lower_data) {
199 lower_data = val;
200 } else if (val > upper_data) {
201 upper_data = val;
202 }
203
204 auto idx = extracted.index[i];
205 if (idx > upper_index) {
206 upper_index = idx;
207 }
208 }
209 };
210
211 if (mat->prefer_rows()) {
212 auto wrk = mat->new_workspace(true);
213 std::vector<T> xbuffer(NC);
214 std::vector<IDX> ibuffer(NC);
215 for (size_t r = 0; r < NR; ++r) {
216 auto extracted = mat->sparse_row(r, xbuffer.data(), ibuffer.data(), wrk.get());
217 update_stats(extracted);
218 }
219 } else {
220 auto wrk = mat->new_workspace(false);
221 std::vector<T> xbuffer(NR);
222 std::vector<IDX> ibuffer(NR);
223 for (size_t c = 0; c < NC; ++c) {
224 auto extracted = mat->sparse_column(c, xbuffer.data(), ibuffer.data(), wrk.get());
225 update_stats(extracted);
226 }
227 }
228
229 // Choosing the types.
230 auto data_type = params.data_type;
231 if (data_type == WriteSparseMatrixToHdf5Parameters::StorageType::AUTOMATIC) {
232 if (non_integer && !params.force_integer) {
233 data_type = WriteSparseMatrixToHdf5Parameters::StorageType::DOUBLE;
234 } else {
235 if (lower_data < 0) {
236 if (HDF5::fits_lower_limit<int8_t>(lower_data) && HDF5::fits_upper_limit<int8_t>(upper_data)) {
237 data_type = WriteSparseMatrixToHdf5Parameters::StorageType::INT8;
238 } else if (HDF5::fits_lower_limit<int16_t>(lower_data) && HDF5::fits_upper_limit<int16_t>(upper_data)) {
239 data_type = WriteSparseMatrixToHdf5Parameters::StorageType::INT16;
240 } else {
241 data_type = WriteSparseMatrixToHdf5Parameters::StorageType::INT32;
242 }
243 } else {
244 if (HDF5::fits_upper_limit<uint8_t>(upper_data)) {
245 data_type = WriteSparseMatrixToHdf5Parameters::StorageType::UINT8;
246 } else if (HDF5::fits_upper_limit<uint16_t>(upper_data)) {
247 data_type = WriteSparseMatrixToHdf5Parameters::StorageType::UINT16;
248 } else {
249 data_type = WriteSparseMatrixToHdf5Parameters::StorageType::UINT32;
250 }
251 }
252 }
253 }
254
255 auto index_type = params.index_type;
256 if (index_type == WriteSparseMatrixToHdf5Parameters::StorageType::AUTOMATIC) {
257 if (HDF5::fits_upper_limit<uint8_t>(upper_index)) {
258 index_type = WriteSparseMatrixToHdf5Parameters::StorageType::UINT8;
259 } else if (HDF5::fits_upper_limit<uint16_t>(upper_index)) {
260 index_type = WriteSparseMatrixToHdf5Parameters::StorageType::UINT16;
261 } else {
262 index_type = WriteSparseMatrixToHdf5Parameters::StorageType::UINT32;
263 }
264 }
265
266 // And then saving it. This time we have no choice but to iterate by the desired dimension.
267 H5::DataSet data_ds = HDF5::create_1d_compressed_hdf5_dataset(location, data_type, params.data_name, non_zeros, params.deflate_level, params.chunk_size);
268 H5::DataSet index_ds = HDF5::create_1d_compressed_hdf5_dataset(location, index_type, params.index_name, non_zeros, params.deflate_level, params.chunk_size);
269 hsize_t count = 0, offset = 0;
270 H5::DataSpace inspace(1, &non_zeros);
271 H5::DataSpace outspace(1, &non_zeros);
272 const auto& dstype = HDF5::define_mem_type<T>();
273 const auto& ixtype = HDF5::define_mem_type<IDX>();
274
275 auto fill_datasets = [&](const SparseRange<T, IDX>& extracted) -> void {
276 count = extracted.number;
277 if (count) {
278 inspace.setExtentSimple(1, &count);
279 outspace.selectHyperslab(H5S_SELECT_SET, &count, &offset);
280 data_ds.write(extracted.value, dstype, inspace, outspace);
281 index_ds.write(extracted.index, ixtype, inspace, outspace);
282 offset += count;
283 }
284 };
285
286 auto layout = params.columnar;
287 if (layout == WriteSparseMatrixToHdf5Parameters::StorageLayout::AUTOMATIC) {
288 if (mat->prefer_rows()) {
289 layout = WriteSparseMatrixToHdf5Parameters::StorageLayout::ROW;
290 } else {
291 layout = WriteSparseMatrixToHdf5Parameters::StorageLayout::COLUMN;
292 }
293 }
294
295 std::vector<hsize_t> ptrs;
296 if (layout == WriteSparseMatrixToHdf5Parameters::StorageLayout::ROW) {
297 ptrs.resize(NR + 1);
298 auto wrk = mat->new_workspace(true);
299 std::vector<T> xbuffer(NC);
300 std::vector<IDX> ibuffer(NC);
301 for (size_t r = 0; r < NR; ++r) {
302 auto extracted = mat->sparse_row(r, xbuffer.data(), ibuffer.data(), wrk.get());
303 fill_datasets(extracted);
304 ptrs[r+1] = ptrs[r] + extracted.number;
305 }
306 } else {
307 ptrs.resize(NC + 1);
308 auto wrk = mat->new_workspace(false);
309 std::vector<T> xbuffer(NR);
310 std::vector<IDX> ibuffer(NR);
311 for (size_t c = 0; c < NC; ++c) {
312 auto extracted = mat->sparse_column(c, xbuffer.data(), ibuffer.data(), wrk.get());
313 fill_datasets(extracted);
314 ptrs[c+1] = ptrs[c] + extracted.number;
315 }
316 }
317
318 // Saving the pointers.
319 hsize_t ptr_len = ptrs.size();
320 H5::DataSet ptr_ds = HDF5::create_1d_compressed_hdf5_dataset(location, H5::PredType::NATIVE_HSIZE, params.ptr_name, ptr_len, params.deflate_level, params.chunk_size);
321 H5::DataSpace ptr_space(1, &ptr_len);
322 ptr_ds.write(ptrs.data(), H5::PredType::NATIVE_HSIZE, ptr_space);
323
324 return;
325}
