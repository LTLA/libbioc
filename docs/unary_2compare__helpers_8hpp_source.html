26template<DelayedCompareOp op_, typename Scalar_>
27bool delayed_compare_actual_sparse(Scalar_ scalar) {
28 if constexpr(op_ == DelayedCompareOp::EQUAL) {
29 return scalar != 0;
30 } else if constexpr(op_ == DelayedCompareOp::GREATER_THAN) {
31 return scalar >= 0;
32 } else if constexpr(op_ == DelayedCompareOp::LESS_THAN) {
33 return scalar <= 0;
34 } else if constexpr(op_ == DelayedCompareOp::GREATER_THAN_OR_EQUAL) {
35 return scalar > 0;
36 } else if constexpr(op_ == DelayedCompareOp::LESS_THAN_OR_EQUAL) {
37 return scalar < 0;
38 } else { // NOT_EQUAL.
39 return scalar == 0;
40 }
