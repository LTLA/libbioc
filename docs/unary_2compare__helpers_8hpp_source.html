19template<CompareOperation op_, typename InputValue_, typename Index_, typename OutputValue_>
20void delayed_compare_run_simple(const InputValue_* input, Index_ length, InputValue_ scalar, OutputValue_* output) {
21 for (Index_ i = 0; i < length; ++i) {
22 if constexpr(std::is_same<InputValue_, OutputValue_>::value) {
23 auto& val = output[i];
24 val = delayed_compare<op_>(val, scalar);
25 } else {
26 output[i] = delayed_compare<op_>(input[i], scalar);
27 }
28 }
29}
