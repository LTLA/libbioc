164 struct SparseBase {
165 /*
166 * There's no need to have a vbuffer here, because we can directly extract into the user-supplied vbuffer if we want the values.
167 * The inner extraction is guaranteed to be no greater than the vbuffer length, as it's a 1:1 mapping between unique indices in the row/column call and the unique internal subset indices.
168 * Temporaries are not a concern because we end up copying everything into sortspace for re-sorting anyway.
169 *
170 * Similarly, an ibuffer is _usually_ unnecessary as we can write directly into the user-supplied ibuffer.
171 * There's only a need to allocate our own ibuffer if (i) the user wants the values but not the indices, and (ii) they want the output to be sorted.
172 * In this case, the user-supplied ibuffer (i) may not be valid but (ii) we still need the indices, so we need to create our ibuffer.
173 */
174
175 SparseBase(size_t bufsize, const WorkspaceOptions& opt) :
176 report_index(sparse_extract_index(opt.sparse_extract_mode)),
177 needs_sort(opt.sparse_ordered_index),
178 ibuffer(opt.sparse_extract_mode == SparseExtractMode::VALUE && needs_sort ? bufsize : 0)
179 {
180 if (needs_sort) {
181 sortspace.reserve(bufsize);
182 }
183 }
184
185 bool report_index;
186 bool needs_sort;
187 std::vector<IDX> ibuffer;
188 std::vector<std::pair<IDX, T> > sortspace;
189 };

207 template<bool WORKROW, class InputWorkspace>
208 SparseRange<T, IDX> reorganize_sparse(size_t i, T* vbuffer, IDX* ibuffer, InputWorkspace* work) const {
209 if (!(work->needs_sort)) {
210 // When we don't need sorting, validity of ibuffer and vbuffer/ should be consistent
211 // with the extraction mode used to construct work->internal.
212 auto raw = extract_sparse<WORKROW>(mat.get(), i, vbuffer, ibuffer, work->internal.get());
213 if (raw.index) {
214 auto icopy = ibuffer;
215 for (size_t i = 0; i < raw.number; ++i, ++icopy) {
216 *icopy = mapping_single[raw.index[i]];
217 }
218 }
219 return raw;
220 }
221
222 // If the workspace's ibuffer is empty, we're either extracting the indices
223 // directly into the user's ibuffer, or we don't need the indices at all.
224 // Either way, it doesn't hurt to use the user's ibuffer.
225 IDX* iin = (work->ibuffer.empty() ? ibuffer : work->ibuffer.data());
226
227 auto raw = extract_sparse<WORKROW>(mat.get(), i, vbuffer, iin, work->internal.get());
228 if (!raw.value && !raw.index) {
229 return raw;
230 }
231
232 auto& sortspace = work->sortspace;
233 sortspace.clear();
234
235 // raw.index is guaranteed to be valid as this point, as the internal workspace
236 // can never be set to VALUE-only if it's requesting sorted output.
237 if (raw.value) {
238 for (size_t i = 0; i < raw.number; ++i) {
239 sortspace.emplace_back(mapping_single[raw.index[i]], raw.value[i]);
240 }
241 } else {
242 for (size_t i = 0; i < raw.number; ++i) {
243 sortspace.emplace_back(mapping_single[raw.index[i]], 0);
244 }
245 }
246
247 std::sort(sortspace.begin(), sortspace.end());
248
249 if (raw.value) {
250 auto vcopy = vbuffer;
251 for (const auto& x : sortspace) {
252 *vcopy = x.second;
253 ++vcopy;
254 }
255 } else {
256 vbuffer = NULL;
257 }
258
259 if (work->report_index) {
260 auto icopy = ibuffer;
261 for (const auto& x : sortspace) {
262 *icopy = x.first;
263 ++icopy;
264 }
265 } else {
266 ibuffer = NULL;
267 }
268
269 return SparseRange<T, IDX>(raw.number, vbuffer, ibuffer);
270 }

354 template<class InputWorkspace>
355 std::vector<IDX> transplant_indices(std::vector<std::pair<V_type, size_t> >& collected, InputWorkspace* work) const {
356 std::sort(collected.begin(), collected.end());
357
358 std::vector<IDX> local;
359 local.reserve(collected.size());
360 if constexpr(!InputWorkspace::sparse) {
361 work->reverse_mapping.resize(collected.size());
362 }
363
364 for (size_t i = 0, end = collected.size(); i < end; ++i) {
365 local.push_back(collected[i].first);
366 if constexpr(!InputWorkspace::sparse) {
367 work->reverse_mapping[collected[i].second] = local.size() - 1;
368 }
369 }
370
371 return local;
372 }
