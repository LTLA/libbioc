31template<typename Index_, class IndexStorage_, class ToIndex_>
32DenseParallelResults<Index_> format_dense_parallel(const IndexStorage_& indices, Index_ len, ToIndex_ to_index) {
33 std::vector<std::pair<Index_, Index_> > collected;
34 collected.reserve(len);
35 for (Index_ i = 0; i < len; ++i) {
36 collected.emplace_back(indices[to_index(i)], i);
37 }
38 std::sort(collected.begin(), collected.end());
39
40 DenseParallelResults<Index_> output;
41 output.sorted.reserve(len);
42 output.permutation.reserve(len);
43 for (const auto& pp : collected) {
44 output.sorted.push_back(pp.first);
45 output.permutation.push_back(pp.second);
46 }
47
48 return output;
49}

133 void initialize(const Matrix<Value_, Index_>* mat, std::vector<Index_> sorted, size_t extent, bool row, MaybeOracle<oracle_, Index_> oracle, Options opt) {
134 needs_value = opt.sparse_extract_value;
135 needs_index = opt.sparse_extract_index;
136 needs_sort = opt.sparse_ordered_index;
137
138 // The conditionals here mirror those in 'fetch',
139 // to self-document the case where each of the temporaries are needed.
140 if (!needs_sort) {
141 if (needs_index) {
142 ; // no 'iholding' required as a user-provided 'ibuffer' should be available.
143 }
144
145 } else if (needs_value) {
146 opt.sparse_extract_index = true;
147 sortspace.reserve(extent);
148 if (needs_index) {
149 ; // no 'iholding' required as a user-provided 'ibuffer' should be available.
150 } else {
151 iholding.resize(extent); // needs 'iholding' as user-provided 'ibuffer' may be NULL.
152 }
153
154 } else if (needs_index) {
155 ; // no 'iholding' required as a user-provided 'ibuffer' should be available.
156 }
157
158 internal = new_extractor<true, oracle_>(mat, row, std::move(oracle), std::move(sorted), opt);

161public:
162 SparseRange<Value_, Index_> fetch(Index_ i, Value_* vbuffer, Index_* ibuffer) {
163 auto input = internal->fetch(i, vbuffer, (iholding.empty() ? ibuffer : iholding.data()));
164
165 // Pointers in 'input' and the 'buffer' pointers may point to the same array,
166 // as we're either just modifiying in place or we're copying to 'sortspace'.
167 if (!needs_sort) {
168 if (needs_index) {
169 for (Index_ i = 0; i < input.number; ++i) {
170 ibuffer[i] = remapping[input.index[i]];
171 }
172 input.index = ibuffer;
173 }
174
175 } else if (needs_value) {
176 // We assume that the indices have already been extracted for sorting
177 // purposes, even if they weren't actually requested.
178 sortspace.clear();
179 for (Index_ i = 0; i < input.number; ++i) {
180 sortspace.emplace_back(remapping[input.index[i]], input.value[i]);
181 }
182 std::sort(sortspace.begin(), sortspace.end());
183
184 auto vcopy = vbuffer;
185 for (const auto& ss : sortspace) {
186 *vcopy = ss.second;
187 ++vcopy;
188 }
189 input.value = vbuffer;
190
191 if (needs_index) {
192 auto icopy = ibuffer;
193 for (const auto& ss : sortspace) {
194 *icopy = ss.first;
195 ++icopy;
196 }
197 input.index = ibuffer;
198 } else {
199 input.index = NULL;
200 }
201
202 } else if (needs_index) {
203 for (Index_ i = 0; i < input.number; ++i) {
204 ibuffer[i] = remapping[input.index[i]];
205 }
206 std::sort(ibuffer, ibuffer + input.number);
207 input.index = ibuffer;
208 }
209
210 return input;
211 }

251 Index_ fulldim = by_row ? mat->nrow() : mat->ncol();
