132 const T_out* cast_dense(size_t i, T_out* buffer, size_t length, InputWorkspace* work) const {
133 if constexpr(same_T_type) {
134 if constexpr(WORKROW) {
135 return ptr->row(i, buffer, work->internal.get());

146 std::copy(out, out + length, buffer);
